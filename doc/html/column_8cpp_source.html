39 std::tuple<QVariant, QVariant, const Item&> Column::multivalue(const Item* item) const {
40  // Transform values into something optimal for sorting
41  // Possibilities: 12, 12.12, 10%, 10.13%, +16%, 12-14, 10/20
42  QVariant sort_first_by;
43  QVariant sort_second_by;
44 
45  QString str = value(*item).toString();
46  QRegularExpressionMatch match;
47 
48  if (str.contains(sort_double_match, &match)) {
49  sort_first_by = match.captured(1).toDouble();
50  } else if (str.contains(sort_two_values, &match)) {
51  if (match.captured(2).startsWith("-")) {
52  sort_first_by = 0.5 * (match.captured(1).toDouble() + match.captured(3).toDouble());
53  } else {
54  sort_first_by = item->PrettyName().c_str();
55  sort_second_by = match.captured(1).toDouble();
56  }
57  } else {
58  sort_first_by = str;
59  sort_second_by = item->PrettyName().c_str();
60  }
61 
62  return std::forward_as_tuple(sort_first_by, sort_second_by, *item);
63 }

274 std::tuple<int, double, const Item&> PriceColumn::multivalue(const Item* item) const {
275  const Buyout &bo = bo_manager_.Get(*item);
276  // forward_as_tuple used to forward item reference properly and avoid ref to temporary
277  // that will be destroyed. We want item reference because we want to sort based on item
278  // object itself and not the pointer. I'm not entirely sure I fully understand
279  // this mechanism, but basically want to avoid a ton of Item copies during sorting
280  // so trying to avoid pass by value (ericsium).
281  return std::forward_as_tuple(bo.currency.AsRank(), bo.value, *item);
282 }
