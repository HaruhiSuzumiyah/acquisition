79 QVariant ItemsModel::data(const QModelIndex &index, int role) const {
80  // Bucket title
81  if (!index.isValid())
82  return QVariant();
83 
84  if (index.internalId() == 0) {
85  if (index.column() > 0)
86  return QVariant();
87 
88  const ItemLocation &location = search_.GetTabLocation(index);
89  if ( role == Qt::CheckStateRole) {
90  if (!location.IsValid())
91  return QVariant();
92  if (bo_manager_.GetRefreshLocked(location))
93  return Qt::PartiallyChecked;
94  return ( bo_manager_.GetRefreshChecked(location) ? Qt::Checked : Qt::Unchecked );
95  }
96  if (role == Qt::DisplayRole) {
97  if (!location.IsValid())
98  return "All Items";
99  QString title(location.GetHeader().c_str());
100  auto const &bo = bo_manager_.GetTab(location.GetUniqueHash());
101  if (bo.IsActive())
102  title += QString(" [%1]").arg(bo.AsText().c_str());
103  return title;
104  }
105  if (location.IsValid() && location.get_type() == ItemLocationType::STASH){
106  if(role == Qt::BackgroundRole){
107  QColor backgroundColor(location.getR(), location.getG(), location.getB());
108  if (backgroundColor.isValid())
109  return backgroundColor;
110  }
111  if(role == Qt::ForegroundRole){
112  QColor backgroundColor(location.getR(), location.getG(), location.getB());
113  return Util::recommendedForegroundTextColor(backgroundColor);
114  }
115 
116  }
117 
118  return QVariant();
119  }
120  auto &column = search_.columns()[index.column()];
121  const Item &item = *search_.bucket(index.parent().row())->item(index.row());
122  if (role == Qt::DisplayRole)
123  return column->value(item);
124  else if (role == Qt::ForegroundRole)
125  return column->color(item);
126  else if (role == Qt::DecorationRole)
127  return column->icon(item);
128  return QVariant();
129 }

144 bool ItemsModel::setData(const QModelIndex &index, const QVariant &value, int role) {
145  if (!index.isValid())
146  return false;
147 
148  if (role == Qt::CheckStateRole) {
149  const ItemLocation &location = search_.GetTabLocation(index);
150  bo_manager_.SetRefreshChecked(location, value.toBool());
151 
152  // It's possible that our tabs can have the same name. Right now we don't have a
153  // way to differentiate these tabs so indicate dataChanged event for each tab with
154  // the same name as the current checked tab so the 'check' is properly updated in
155  // the layout
156  std::string target_hash = location.GetUniqueHash();
157  auto row_count = rowCount();
158  for (int i = 0; i < row_count; ++i) {
159  auto match_index = this->index(i);
160  if (search_.GetTabLocation(match_index).GetUniqueHash() == target_hash)
161  emit dataChanged(match_index,match_index);
162  }
163 
164  return true;
165  }
166  return false;
167 }

169 void ItemsModel::sort(int column, Qt::SortOrder order)
170 {
171  // Ignore sort requests if we're already sorted
172  if (sorted_ && (sort_column_ == column) && (sort_order_ == order))
173  return;
174 
175  QLOG_DEBUG() << "Sorting";
176  sort_order_ = order;
177  sort_column_ = column;
178 
179  auto &column_obj = search_.columns()[column];
180  for (const auto &bucket: search_.buckets()) {
181  bucket->Sort(*column_obj, order);
182  }
183  layoutChanged();
184  SetSorted(true);
185 }
