124 bool ModsFilter::Matches(const std::shared_ptr<Item> &item, FilterData *data) {
125  for (auto &mod : data->mod_data) {
126  if (mod.mod.empty())
127  continue;
128  const ModTable &mod_table = item->mod_table();
129  if (!mod_table.count(mod.mod))
130  return false;
131  double value = mod_table.at(mod.mod);
132  if (mod.min_filled && value < mod.min)
133  return false;
134  if (mod.max_filled && value > mod.max)
135  return false;
136  }
137  return true;
138 }
