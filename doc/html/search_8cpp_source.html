99 const std::unique_ptr<Bucket> &Search::bucket(int row) const {
100  auto const &active_buckets = (current_mode_ == ByTab) ? buckets_:bucket_;
101 
102  if (row >= 0) {
103  std::vector<std::unique_ptr<Bucket>>::size_type row_t = (size_t) row; // Assumes int max() always able to fit in unsigned long long
104 
105  if (row_t < active_buckets.size()) {
106  return active_buckets[row_t];
107  }
108  }
109 
110  QMessageBox::critical(nullptr, "Fatal Error", QString("Bucket row out of bounds: ") +
111  QString::number(row) + " bucket size: " + QString::number(active_buckets.size()) +
112  " mode:" + QString::number(current_mode_) +
113  ". Program will abort.");
114  abort();

118 void Search::FilterItems(const Items &items) {
119  // If we're just changing tabs we don't need to update anything
120  if (refresh_reason_ == RefreshReason::TabChanged)
121  return;
122 
123  QLOG_DEBUG() << "FilterItems: reason(" << refresh_reason_ << ")";
124  items_.clear();
125  for (const auto &item : items) {
126  bool matches = true;
127  for (auto &filter : filters_)
128  if (!filter->Matches(item)) {
129  matches = false;
130  break;
131  }
132  if (matches)
133  items_.push_back(item);
134  }
135 
136  UpdateItemCounts(items);
137 
138  // Single bucket with null location is used to view all items at once
139  bucket_.clear();
140  bucket_.push_back(std::make_unique<Bucket>(ItemLocation()));
141 
142  std::map<ItemLocation, std::unique_ptr<Bucket>> bucketed_tabs;
143  for (const auto &item : items_) {
144  ItemLocation location = item->location();
145  if (!bucketed_tabs.count(location))
146  bucketed_tabs[location] = std::make_unique<Bucket>(location);
147  bucketed_tabs[location]->AddItem(item);
148  bucket_.front()->AddItem(item);
149  }
150 
151  // We need to add empty tabs here as there are no items to force their addition
152  // But only do so if no filters are active as we want to hide empty tabs when
153  // filtering
154  if (!IsAnyFilterActive()) {
155  for (auto &location: bo_manager_.GetStashTabLocations())
156  if (!bucketed_tabs.count(location)) {
157  bucketed_tabs[location] = std::make_unique<Bucket>(location);
158  }
159  }
160 
161  buckets_.clear();
162  for (auto &element : bucketed_tabs)
163  buckets_.push_back(std::move(element.second));
164 
165  // Let the model know that current sort order has been invalidated
166  model_->SetSorted(false);
167 }

177 ItemLocation Search::GetTabLocation(const QModelIndex & index) const {
178  if (!index.isValid())
179  return ItemLocation();
180 
181  if (index.internalId() > 0) {
182  // If index represents an item, get location from item as view may be on 'item' view
183  // where bucket location doesn't match items location
184  return bucket(index.parent().row())->item(index.row())->location();
185  } else {
186  // Otherwise index represents a tab already, get location from there
187  return bucket(index.row())->location();
188  }
189 }

221 void Search::SaveViewProperties() {
222  expanded_property_.clear();
223  auto rowCount = model_->rowCount();
224  for (int row = 0; row < rowCount; ++row) {
225  QModelIndex index = model_->index( row, 0, QModelIndex());
226  if (index.isValid() && view_->isExpanded(index)) {
227  expanded_property_.insert(bucket(row)->location().GetHeader());
228  }
229  }

232 void Search::RestoreViewProperties() {
233  if (!expanded_property_.empty()) {
234  auto rowCount = model_->rowCount();
235  for (int row = 0; row < rowCount; ++row) {
236  QModelIndex index = model_->index( row, 0, QModelIndex());
237  // Block signals else columns will be resized on every expand which can be super slow.
238  view_->blockSignals(true);
239  if (expanded_property_.count(bucket(row)->location().GetHeader())) {
240  view_->expand(index);
241  }
242  view_->blockSignals(false);
243  }
244  }
245 }
