174 bool RaritySearchFilter::Matches(const std::shared_ptr<Item> &item, FilterData *data) {
175  if (data->text_query == "") {
176  return true;
177  }
178  switch (item->frameType()) {
179  case FRAME_TYPE_NORMAL:
180  return (data->text_query == "Normal");
181  case FRAME_TYPE_MAGIC:
182  return (data->text_query == "Magic");
183  case FRAME_TYPE_RARE:
184  return (data->text_query == "Rare");
185  case FRAME_TYPE_UNIQUE:
186  return (data->text_query == "Unique");
187  case FRAME_TYPE_RELIC:
188  return (data->text_query == "Unique (Relic)");
189  default:
190  return false;
191  }
192 }

274 bool MinMaxFilter::Matches(const std::shared_ptr<Item> &item, FilterData *data) {
275  if (IsValuePresent(item)) {
276  double value = GetValue(item);
277  if (data->min_filled && data->min > value)
278  return false;
279  if (data->max_filled && data->max < value)
280  return false;
281  return true;
282  } else {
283  return !data->max_filled && !data->min_filled;
284  }

392  if (!data->r_filled && !data->g_filled && !data->b_filled)

401  const ItemSocketGroup &sockets = item->sockets();
402  return Check(need_r, need_g, need_b, sockets.r, sockets.g, sockets.b, sockets.w);
