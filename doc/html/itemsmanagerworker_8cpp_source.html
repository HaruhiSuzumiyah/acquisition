78 void ItemsManagerWorker::Init() {
79  tabs_.clear();
80  tab_id_index_.clear();
81 
82  //Get cached tabs (item tabs not search tabs)
83  for(ItemLocationType type : {ItemLocationType::STASH, ItemLocationType::CHARACTER}){
84  std::string tabs = data_.GetTabs(type);
85  tabs_signature_ = CreateTabsSignatureVector(tabs);
86  if (tabs.size() != 0) {
87  rapidjson::Document doc;
88  if (doc.Parse(tabs.c_str()).HasParseError()) {
89  QLOG_ERROR() << "Malformed tabs data:" << tabs.c_str() << "The error was"
90  << rapidjson::GetParseError_En(doc.GetParseError());
91  continue;
92  }
93  for (auto &tab : doc) {
94  //constructor values to fill in
95  int index;
96  std::string tabUniqueId, name;
97  int r, g, b;
98 
99  if(type == ItemLocationType::STASH){
100  if(tab_id_index_.count(tab["id"].GetString())){
101  continue;
102  }
103  if (!tab.HasMember("n") || !tab["n"].IsString()) {
104  QLOG_ERROR() << "Malformed tabs data:" << tabs.c_str() << "Tab doesn't contain its name (field 'n').";
105  continue;
106  }
107 
108  index = tab["i"].GetInt();
109  tabUniqueId = tab["id"].GetString();
110  name = tab["n"].GetString();
111  r = tab["colour"]["r"].GetInt();
112  g = tab["colour"]["g"].GetInt();
113  b = tab["colour"]["b"].GetInt();
114  } else {
115  if(tab_id_index_.count(tab["name"].GetString())){
116  continue;
117  }
118 
119  if(tab.HasMember("i"))
120  index = tab["i"].GetInt();
121  else
122  index = tabs_.size();
123 
124  tabUniqueId = tab["name"].GetString();
125  name = tab["name"].GetString();
126  r = 0;
127  g = 0;
128  b = 0;
129  }
130 
131  ItemLocation loc(index, tabUniqueId, name, type, r, g, b);
132  loc.set_json(tab, doc.GetAllocator());
133  tabs_.push_back(loc);
134  tab_id_index_.insert(loc.get_tab_uniq_id());
135  }
136  }
137  }
138 
139  items_.clear();
140 
141  //Get cached items
142  for (auto tab : tabs_){
143  std::string items = data_.GetItems(tab);
144  if (items.size() != 0) {
145  rapidjson::Document doc;
146  doc.Parse(items.c_str());
147  for (auto item = doc.Begin(); item != doc.End(); ++item)
148  items_.push_back(std::make_shared<Item>(*item, tab));
149  }
150  }
151 
152  //let ItemManager know that the retrieval of cached items/tabs has been completed (calls ItemsManager::OnItemsRefreshed method)
153  emit ItemsRefreshed(items_, tabs_, true);
154 }

156 void ItemsManagerWorker::Update(TabSelection::Type type, const std::vector<ItemLocation> &locations) {
157  if (updating_) {
158  QLOG_WARN() << "ItemsManagerWorker::Update called while updating";
159  return;
160  }
161 
162  selected_tabs_.clear();
163  for (auto const &tab: locations) {
164  selected_tabs_.insert(tab.get_tab_uniq_id());
165  }
166 
167  tab_selection_ = type;
168 
169  QLOG_DEBUG() << "Updating" << tab_selection_ << "stash tabs";
170  updating_ = true;
171 
172  cancel_update_ = false;
173  // remove all mappings (from previous requests)
174  if (signal_mapper_)
175  delete signal_mapper_;
176  signal_mapper_ = new QSignalMapper;
177  // remove all pending requests
178  queue_ = std::queue<ItemsRequest>();
179  queue_id_ = 0;
180  replies_.clear();
181  items_.clear();
182  tabs_as_string_ = "";
183  selected_character_ = "";
184 
185  // first, download the main page because it's the only way to know which character is selected
186  QNetworkReply *main_page = network_manager_.get(Request(QUrl(kMainPage), ItemLocation(), TabCache::Refresh));
187  connect(main_page, &QNetworkReply::finished, this, &ItemsManagerWorker::OnMainPageReceived);
188 }

190 void ItemsManagerWorker::OnMainPageReceived() {
191  QNetworkReply *reply = qobject_cast<QNetworkReply *>(QObject::sender());
192 
193  if (reply->error()) {
194  QLOG_WARN() << "Couldn't fetch main page: " << reply->url().toDisplayString() << " due to error: " << reply->errorString();
195  } else {
196  std::string page(reply->readAll().constData());
197 
198  selected_character_ = Util::FindTextBetween(page, "C({\"name\":\"", "\",\"league");
199  if (selected_character_.empty()) {
200  QLOG_WARN() << "Couldn't extract currently selected character name from GGG homepage (maintenence?) Text was: " << page.c_str();
201  }
202  }
203 
204  // now get character list
205  QNetworkReply *characters = network_manager_.get(Request(QUrl(kGetCharactersUrl), ItemLocation(), TabCache::Refresh));
206  connect(characters, &QNetworkReply::finished, this, &ItemsManagerWorker::OnCharacterListReceived);
207 
208  reply->deleteLater();
209 }

211 void ItemsManagerWorker::OnCharacterListReceived() {
212  QNetworkReply *reply = qobject_cast<QNetworkReply *>(QObject::sender());
213  QByteArray bytes = reply->readAll();
214  rapidjson::Document doc;
215  doc.Parse(bytes.constData());
216 
217  if (reply->error()) {
218  QLOG_WARN() << "Couldn't fetch character list: " << reply->url().toDisplayString()
219  << " due to error: " << reply->errorString() << " Aborting update.";
220  updating_ = false;
221  return;
222  } else {
223  if (doc.HasParseError() || !doc.IsArray()) {
224  QLOG_ERROR() << "Received invalid reply instead of character list. The reply was: "
225  << bytes.constData();
226  if (doc.HasParseError()) {
227  QLOG_ERROR() << "The error was" << rapidjson::GetParseError_En(doc.GetParseError());
228  }
229  QLOG_ERROR() << "";
230  QLOG_ERROR() << "(Maybe you need to log in to the website manually and accept new Terms of Service?)";
231  updating_ = false;
232  return;
233  }
234  }
235 
236  QLOG_DEBUG() << "Received character list, there are" << doc.Size() << "characters";
237 
238  //clear out ItemLocationType::CHARACTER tabs from tabs_
239  std::vector<ItemLocation> tmpLocs = tabs_;
240  for(auto &tab : tmpLocs){
241  if(tab.get_type() == ItemLocationType::CHARACTER){
242  tabs_.erase(std::remove(tabs_.begin(), tabs_.end(), tab), tabs_.end());
243  tab_id_index_.erase(tab.get_tab_uniq_id());
244  }
245  }
246 
247  auto char_count = 0;
248  for (auto &character : doc) {
249  if (!character.HasMember("league") || !character.HasMember("name") || !character["league"].IsString() || !character["name"].IsString()) {
250  QLOG_ERROR() << "Malformed character entry, the reply is most likely invalid" << bytes.constData();
251  continue;
252  }
253  if (character["league"].GetString() == league_) {
254  char_count++;
255  std::string name = character["name"].GetString();
256  ItemLocation location;
257  location.set_type(ItemLocationType::CHARACTER);
258  location.set_character(name);
259  location.set_json(character, doc.GetAllocator());
260  location.set_tab_id(tabs_.size());
261  tabs_.push_back(location);
262  //Queue request for items on character in character's stash
263  QueueRequest(MakeCharacterRequest(name, location), location);
264 
265  //Queue request for jewels in character's passive tree
266  QueueRequest(MakeCharacterPassivesRequest(name, location), location);
267  }
268  }
269  CurrentStatusUpdate status;
270  status.state = ProgramState::CharactersReceived;
271  status.total = char_count;
272 
273  emit StatusUpdate(status);
274 
275  if (char_count == 0) {
276  updating_ = false;
277  return;
278  }
279 
280  // Fetch a single tab and also request tabs list. We can fetch any tab here with tabs list
281  // appended, so prefer one that the user has already 'checked'. Default to index '1' which is
282  // first user visible tab.
283  first_fetch_tab_ = "";
284  ItemLocation tabToReq;
285  if (tab_selection_ == TabSelection::Checked) {
286  for (auto const &tab : tabs_) {
287  if (bo_manager_.GetRefreshChecked(tab)) {
288  first_fetch_tab_ = tab.get_tab_uniq_id();
289  tabToReq = tab;
290  break;
291  }
292  }
293  }
294  // If we're refreshing a manual selection of tabs choose one of them to save a tab fetch
295  if (tab_selection_ == TabSelection::Selected) {
296  for (auto const &tab : tabs_) {
297  if (selected_tabs_.count(tab.get_tab_uniq_id())) {
298  first_fetch_tab_ = tab.get_tab_uniq_id();
299  tabToReq = tab;
300  break;
301  }
302  }
303  }
304 
305  QNetworkReply *first_tab = network_manager_.get(MakeTabRequest(tabToReq.get_tab_id(), ItemLocation(), true, true));
306  connect(first_tab, SIGNAL(finished()), this, SLOT(OnFirstTabReceived()));
307  reply->deleteLater();
308 }

310 QNetworkRequest ItemsManagerWorker::MakeTabRequest(int tab_index, const ItemLocation &location, bool tabs, bool refresh) {
311  QUrlQuery query;
312  query.addQueryItem("league", league_.c_str());
313  query.addQueryItem("tabs", tabs ? "1" : "0");
314  query.addQueryItem("tabIndex", std::to_string(tab_index).c_str());
315  query.addQueryItem("accountName", account_name_.c_str());
316 
317  QUrl url(kStashItemsUrl);
318  url.setQuery(query);
319 
320  // If refresh is explicity request then force unconditionally
321  TabCache::Flags flags = (refresh) ? TabCache::Refresh : TabCache::None;
322 
323  return Request(url, location, flags);
324 }

348 QNetworkRequest ItemsManagerWorker::Request(QUrl url, const ItemLocation &location, TabCache::Flags flags) {
349  switch (tab_selection_) {
350  case TabSelection::All:
351  flags |= TabCache::Refresh;
352  break;
353  case TabSelection::Checked:
354  if (!location.IsValid() || bo_manager_.GetRefreshChecked(location))
355  flags |= TabCache::Refresh;
356  break;
357  case TabSelection::Selected:
358  if (!location.IsValid() || selected_tabs_.count(location.get_tab_uniq_id()))
359  flags |= TabCache::Refresh;
360  break;
361  }
362  return tab_cache_->Request(url, flags);
363 }

365 void ItemsManagerWorker::QueueRequest(const QNetworkRequest &request, const ItemLocation &location) {
366  QLOG_DEBUG() << "Queued (" << queue_id_ + 1 << ") -- " << location.GetHeader().c_str();

374 void ItemsManagerWorker::FetchItems(int limit) {
375  std::string tab_titles;
376  int count = std::min(limit, static_cast<int>(queue_.size()));
377  for (int i = 0; i < count; ++i) {
378  ItemsRequest request = queue_.front();
379  queue_.pop();
380 
381  QNetworkReply *fetched = network_manager_.get(request.network_request);
382  signal_mapper_->setMapping(fetched, request.id);
383  connect(fetched, SIGNAL(finished()), signal_mapper_, SLOT(map()));
384 
385  ItemsReply reply;
386  reply.network_reply = fetched;
387  reply.request = request;
388  replies_[request.id] = reply;
389 
390  tab_titles += request.location.GetHeader() + " ";
391  }
392  QLOG_DEBUG() << "Created" << count << "requests:" << tab_titles.c_str();
393  requests_needed_ = count;
394  requests_completed_ = 0;
395  cached_requests_completed_ = 0;
396 }

398 void ItemsManagerWorker::OnFirstTabReceived() {
399  QNetworkReply *reply = qobject_cast<QNetworkReply *>(QObject::sender());
400  QByteArray bytes = reply->readAll();
401  rapidjson::Document doc;
402  doc.Parse(bytes.constData());
403 
404  if (!doc.IsObject()) {
405  QLOG_ERROR() << "Can't even fetch first tab. Failed to update items.";
406  updating_ = false;
407  return;
408  }
409 
410  if (doc.HasMember("error")) {
411  QLOG_ERROR() << "Aborting update since first fetch failed due to 'error': " << Util::RapidjsonSerialize(doc["error"]).c_str();
412  updating_ = false;
413  return;
414  }
415 
416  if (!doc.HasMember("tabs") || doc["tabs"].Size() == 0) {
417  QLOG_WARN() << "There are no tabs, this should not happen, bailing out.";
418  updating_ = false;
419  return;
420  }
421 
422  tabs_as_string_ = Util::RapidjsonSerialize(doc["tabs"]);
423  tabs_signature_ = CreateTabsSignatureVector(tabs_as_string_);
424 
425  QLOG_DEBUG() << "Received tabs list, there are" << doc["tabs"].Size() << "tabs";
426 
427  std::set<std::string> old_tab_headers;
428  for (auto const &tab: tabs_) {
429  // Remember old tab headers before clearing tabs
430  old_tab_headers.insert(tab.GetHeader());
431  }
432 
433  //clear out ItemLocationType::STASH tabs from tabs_
434  std::vector<ItemLocation> tmpLocs = tabs_;
435  for(auto &tab : tmpLocs){
436  if(tab.get_type() == ItemLocationType::STASH){
437  tabs_.erase(std::remove(tabs_.begin(), tabs_.end(), tab), tabs_.end());
438  tab_id_index_.erase(tab.get_tab_uniq_id());
439  }
440  }
441 
442  // Create tab location objects
443  for (auto &tab : doc["tabs"]) {
444  std::string label = tab["n"].GetString();
445  auto index = tab["i"].GetInt();
446  // Ignore hidden locations
447  if (!doc["tabs"][index].HasMember("hidden") || !doc["tabs"][index]["hidden"].GetBool()){
448  int r = tab["colour"]["r"].GetInt();
449  int g = tab["colour"]["g"].GetInt();
450  int b = tab["colour"]["b"].GetInt();
451 
452  ItemLocation loc(index, tab["id"].GetString(), label, ItemLocationType::STASH, r, g, b);
453  loc.set_json(tab, doc.GetAllocator());
454  tabs_.push_back(loc);
455  tab_id_index_.insert(tab["id"].GetString());
456  }
457  }
458 
459  // Queue requests for Stash tabs
460  for (auto const &tab: tabs_) {
461  bool refresh = false;
462 
463  // Force refreshes for any tabs that were moved or renamed regardless of what user
464  // requests for refresh.
465  if (!old_tab_headers.count(tab.GetHeader())) {
466  QLOG_DEBUG() << "Forcing refresh of moved or renamed tab: " << tab.GetHeader().c_str();
467  refresh = true;
468  }
469  if(tab.get_type() == ItemLocationType::STASH)
470  QueueRequest(MakeTabRequest(tab.get_tab_id(), tab, true, refresh), tab);
471  }
472 
473  total_needed_ = queue_.size();
474  total_completed_ = 0;
475  total_cached_ = reply->attribute(QNetworkRequest::SourceIsFromCacheAttribute).toBool() ? 1:0;
476 
477  FetchItems(kThrottleRequests - 1);
478 
479  connect(signal_mapper_, SIGNAL(mapped(int)), this, SLOT(OnTabReceived(int)));
480  reply->deleteLater();
481 }

503 void ItemsManagerWorker::OnTabReceived(int request_id) {
504  if (!replies_.count(request_id)) {
505  QLOG_WARN() << "Received a reply for request" << request_id << "that was not requested.";
506  return;
507  }
508 
509  ItemsReply reply = replies_[request_id];
510 
511  bool reply_from_cache = reply.network_reply->attribute(QNetworkRequest::SourceIsFromCacheAttribute).toBool();
512 
513  if (reply_from_cache) {
514  QLOG_DEBUG() << "Received a cached reply for" << reply.request.location.GetHeader().c_str();
515  ++cached_requests_completed_;
516  ++total_cached_;
517  } else {
518  QLOG_DEBUG() << "Received a reply for" << reply.request.location.GetHeader().c_str();
519  }
520 
521  QByteArray bytes = reply.network_reply->readAll();
522  rapidjson::Document doc;
523  doc.Parse(bytes.constData());
524 
525  bool error = false;
526  if (!doc.IsObject()) {
527  QLOG_WARN() << request_id << "got a non-object response";
528  error = true;
529  } else if (doc.HasMember("error")) {
530  // this can happen if user is browsing stash in background and we can't know about it
531  QLOG_WARN() << request_id << "got 'error' instead of stash tab contents: " << Util::RapidjsonSerialize(doc["error"]).c_str();
532  error = true;
533  }
534 
535  // We index expected tabs and their locations as part of the first fetch. It's possible for users
536  // to move or rename tabs during the update which will result in the item data being out-of-sync with
537  // expected index/tab name map. We need to detect this case and abort the update.
538  if (!cancel_update_ && !error && (reply.request.location.get_type() == ItemLocationType::STASH)) {
539  if (!doc.HasMember("tabs") || doc["tabs"].Size() == 0) {
540  QLOG_ERROR() << "Full tab information missing from stash tab fetch. Cancelling update. Full fetch URL: "
541  << reply.request.network_request.url().toDisplayString();
542  cancel_update_ = true;
543  } else {
544  std::string tabs_as_string = Util::RapidjsonSerialize(doc["tabs"]);
545  auto tabs_signature_current = CreateTabsSignatureVector(tabs_as_string);
546 
547  auto tab_id = reply.request.location.get_tab_id();
548  if (tabs_signature_[tab_id] != tabs_signature_current[tab_id]) {
549  if (reply_from_cache) {
550  // Here we unexpectedly are seeing a cached document that is out-of-sync with current tab state
551  // This is not fatal but unexpected as we shouldn't get here if everything else is done right.
552  // If we do see, set 'error' condition which causes us to flush from catch and re-fetch from server.
553  QLOG_WARN() << "Unexpected hit on stale cached tab. Flushing and re-fetching request: "
554  << reply.request.network_request.url().toDisplayString();
555  error = true;
556  // Isn't really cached since we're erroring out and replaying so fix up stats
557  total_cached_--;
558  } else {
559  std::string reason;
560  if (tabs_signature_current.size() != tabs_signature_.size())
561  reason += "[Tab size mismatch:" + std::to_string(tabs_signature_current.size()) + " != "
562  + std::to_string(tabs_signature_.size()) + "]";
563 
564  auto &x = tabs_signature_current[tab_id];
565  auto &y = tabs_signature_[tab_id];
566  reason += "[tab_index=" + std::to_string(tab_id) + "/" + std::to_string(tabs_signature_current.size()) + "(#" + std::to_string(tab_id+1) + ")]";
567  if (x.first != y.first)
568  reason += "[name:" + x.first + " != " + y.first + "]";
569  if (x.second != y.second)
570  reason += "[id:" + x.second + " != " + y.second + "]";
571 
572  QLOG_ERROR() << "You renamed or re-ordered tabs in game while acquisition was in the middle of the update,"
573  << " aborting to prevent synchronization problems and pricing data loss. Mismatch reason(s) -> "
574  << reason.c_str() << ". For request: " << reply.request.network_request.url().toDisplayString();
575  cancel_update_ = true;
576  }
577  }
578  }
579  }
580 
581  // re-queue a failed request
582  if (error) {
583  // We can 'cache' error response document so make sure we remove it
584  // before reque
585  tab_cache_->remove(reply.request.network_request.url());
586  QueueRequest(reply.request.network_request, reply.request.location);
587  }
588 
589  ++requests_completed_;
590 
591  if (!error)
592  ++total_completed_;
593 
594  bool throttled = false;
595 
596  if (requests_completed_ == requests_needed_) {
597  if (cancel_update_) {
598  updating_ = false;
599  } else if (queue_.size() > 0) {
600  if (cached_requests_completed_ > 0) {
601  // We basically don't want cached requests to count against throttle limit
602  // so if we did get any cached requests fetch up to that number without a
603  // large delay
604  QTimer::singleShot(1, [&]() { FetchItems(cached_requests_completed_); });
605  } else {
606  throttled = true;
607  QLOG_DEBUG() << "Sleeping one minute to prevent throttling.";
608  QTimer::singleShot(kThrottleSleep * 1000, this, SLOT(FetchItems()));
609  }
610  }

613  status.state = throttled ? ProgramState::ItemsPaused : ProgramState::ItemsReceive;
614  status.progress = total_completed_;
615  status.total = total_needed_;
616  status.cached = total_cached_;
617  if (total_completed_ == total_needed_)
618  status.state = ProgramState::ItemsCompleted;
619  if (cancel_update_)

622 
623  if (error)

625 
626  ParseItems(&doc["items"], reply.request.location, doc.GetAllocator());

630  // move items around in a tab and we get them in a different order. For
631  // consistency we want to present the tab data in a deterministic way to the rest
632  // of the application. Especially so we don't try to update shop when nothing actually
633  // changed. So sort items_ here before emitting and then generate
634  // item list as strings.
635

638  });

644  }
645

647  std::map<ItemLocation, QStringList> itemsPerLoc;
648  for (auto const &item: items_) {
649  itemsPerLoc[item->location()].push_back(item->json().c_str());
650  }
651

653  std::map<ItemLocationType, QStringList>::iterator tabIter;
654  for ( tabIter = tabsPerType.begin(); tabIter != tabsPerType.end(); tabIter++ ){
655  data_.SetTabs(tabIter->first, std::string("[") + tabIter->second.join(",").toStdString() + "]");
656  }
657 
658  //save items
659  std::map<ItemLocation, QStringList>::iterator itemIter;
660  for ( itemIter = itemsPerLoc.begin(); itemIter != itemsPerLoc.end(); itemIter++ )
661  {
662  data_.SetItems(itemIter->first, std::string("[") + itemIter->second.join(",").toStdString() + "]");
663  }
664 
665  // all requests completed
666  emit ItemsRefreshed(items_, tabs_, false);
667 
668  updating_ = false;
669  QLOG_DEBUG() << "Finished updating stash.";
670 
671  // if we're at the verge of getting throttled, sleep so we don't
672  if (requests_completed_ == kThrottleRequests)
673  QTimer::singleShot(kThrottleSleep, this, SLOT(PreserveSelectedCharacter()));
674  else
675  PreserveSelectedCharacter();
676  }
677 
678  reply.network_reply->deleteLater();
