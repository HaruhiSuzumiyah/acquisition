205 void BuyoutManager::Set(const Item &item, const Buyout &buyout) {
206  auto it = buyouts_.lower_bound(item.hash());
207  if (it != buyouts_.end() && !(buyouts_.key_comp()(item.hash(), it->first))) {
208  // Entry exists - we don't want to update if buyout is equal to existing
209  if (buyout != it->second) {
210  save_needed_ = true;
211  it->second = buyout;
212  }
213  } else {
214  save_needed_ = true;
215  buyouts_.insert(it, {item.hash(), buyout});
216  }
217 }

249 void BuyoutManager::CompressTabBuyouts() {
250  // When tabs are renamed we end up with stale tab buyouts that aren't deleted.
251  // This function is to remove buyouts associated with tab names that don't
252  // currently exist.
253  std::set<std::string> tmp;
254  for (auto const& loc: tabs_)
255  tmp.insert(loc.GetUniqueHash());
256 
257  for (auto it = tab_buyouts_.begin(), ite = tab_buyouts_.end(); it != ite;) {
258  if (tmp.count(it->first) == 0) {
259  save_needed_ = true;
260  it = tab_buyouts_.erase(it);
261  } else {
262  ++it;
263  }
264  }
265 }

267 void BuyoutManager::CompressItemBuyouts(const Items &items) {
268  // When items are moved between tabs or deleted their buyouts entries remain
269  // This function looks at buyouts and makes sure there is an associated item
270  // that exists
271  std::set<std::string> tmp;
272  for (auto const &item_sp: items) {
273  const Item & item= *item_sp;
274  tmp.insert(item.hash());
275  }
276 
277  for (auto it = buyouts_.cbegin(); it != buyouts_.cend();) {
278  if (tmp.count(it->first) == 0) {
279 
280  buyouts_.erase(it++);
281  } else {
282  ++it;
283  }

319 std::string BuyoutManager::Serialize(const std::map<std::string, Buyout> &buyouts) {
320  rapidjson::Document doc;
321  doc.SetObject();
322  auto &alloc = doc.GetAllocator();
323 
324  for (auto &bo : buyouts) {
325  const Buyout &buyout = bo.second;
326  if (!buyout.IsSavable())
327  continue;
328  rapidjson::Value item(rapidjson::kObjectType);
329  item.AddMember("value", buyout.value, alloc);
330 
331  if (!buyout.last_update.isNull()){
332  item.AddMember("last_update", buyout.last_update.toTime_t(), alloc);
333  }else{
334  // If last_update is null, set as the actual time
335  item.AddMember("last_update", QDateTime::currentDateTime().toTime_t(), alloc);
336  }
337 
338  Util::RapidjsonAddConstString(&item, "type", buyout.BuyoutTypeAsTag(), alloc);
339  Util::RapidjsonAddConstString(&item, "currency", buyout.CurrencyAsTag(), alloc);
340  Util::RapidjsonAddConstString(&item, "source", buyout.BuyoutSourceAsTag(), alloc);
341 
342  item.AddMember("inherited", buyout.inherited, alloc);
343 
344  rapidjson::Value name(bo.first.c_str(), alloc);
345  doc.AddMember(name, item, alloc);
346  }
347 
348  return Util::RapidjsonSerialize(doc);
349 }

351 void BuyoutManager::Deserialize(const std::string &data, std::map<std::string, Buyout> *buyouts) {
352  buyouts->clear();
353 
354  // if data is empty (on first use) we shouldn't make user panic by showing ERROR messages
355  if (data.empty())
356  return;
357 
358  rapidjson::Document doc;
359  if (doc.Parse(data.c_str()).HasParseError()) {
360  QLOG_ERROR() << "Error while parsing buyouts.";
361  QLOG_ERROR() << rapidjson::GetParseError_En(doc.GetParseError());
362  return;
363  }
364  if (!doc.IsObject())
365  return;
366  for (auto itr = doc.MemberBegin(); itr != doc.MemberEnd(); ++itr) {
367  auto &object = itr->value;
368  const std::string &name = itr->name.GetString();
369  Buyout bo;
370 
371  bo.currency = Currency::FromTag(object["currency"].GetString());
372  bo.type = Buyout::TagAsBuyoutType(object["type"].GetString());
373  bo.value = object["value"].GetDouble();
374  if (object.HasMember("last_update")){
375  bo.last_update = QDateTime::fromTime_t(object["last_update"].GetInt());
376  }
377  if (object.HasMember("source")){
378  bo.source = Buyout::TagAsBuyoutSource(object["source"].GetString());
379  }
380  bo.inherited = false;
381  if (object.HasMember("inherited"))
382  bo.inherited = object["inherited"].GetBool();
383  (*buyouts)[name] = bo;
384  }
385 }

461 Buyout BuyoutManager::StringToBuyout(std::string format) {
462  // Parse format string and initialize buyout object, if string does not match any known format
463  // then the buyout object will not be valid (IsValid will return false).
464  std::regex exp("(~\\S+)\\s+(\\d+\\.?\\d*)\\s+(\\w+)");
465 
466  std::smatch sm;
467 
468  Buyout tmp;
469  // regex_search allows for stuff before ~ and after currency type. We only want to honor the formats
470  // that POE trade also accept so this may need to change if it's too generous
471  if (std::regex_search(format,sm,exp)) {
472  tmp.type = StringToBuyoutType(sm[1]);
473  tmp.value = QVariant(sm[2].str().c_str()).toDouble();
474  tmp.currency = StringToCurrencyType(sm[3]);
475  tmp.source = BUYOUT_SOURCE_GAME;
476  tmp.last_update = QDateTime::currentDateTime();
477  }
478  return tmp;
479 }

481 void BuyoutManager::MigrateItem(const Item &item) {
482  std::string old_hash = item.old_hash();
483  std::string hash = item.hash();
484  auto it = buyouts_.find(old_hash);
485  auto new_it = buyouts_.find(hash);
486  if (it != buyouts_.end() && (new_it == buyouts_.end() || new_it->second.source != BUYOUT_SOURCE_MANUAL)) {
487  buyouts_[hash] = it->second;
488  buyouts_.erase(it);
489  save_needed_ = true;
490  }
491 }
