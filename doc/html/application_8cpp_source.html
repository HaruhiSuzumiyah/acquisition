74 void Application::SaveDbOnNewVersion() {
75  //If user updated from a 0.5c db to a 0.5d, db exists but no "version" in it
76  std::string version = data_->Get("version", "0.5c");
77  // We call this just after login, so we didn't pulled tabs for the first time ; so "tabs" shouldn't exist in the DB
78  // This way we don't create an useless data_save_version folder on the first time you run acquisition
79  bool first_start = data_->Get("tabs", "first_time") == "first_time";
80  if (version != VERSION_NAME && !first_start) {
81  QString data_path = Filesystem::UserDir().c_str() + QString("/data");
82  QString save_path = data_path + "_save_" + version.c_str();
83  QDir src(data_path);
84  QDir dst(save_path);
85  if (!dst.exists())
86  QDir().mkpath(dst.path());
87  for (auto name : src.entryList()) {
88  QFile::copy(data_path + QDir::separator() + name, save_path + QDir::separator() + name);
89  }
90  QLOG_INFO() << "I've created the folder " << save_path << "in your acquisition folder, containing a save of all your data";
91  }
92  data_->Set("version", VERSION_NAME);
93 
94 }
