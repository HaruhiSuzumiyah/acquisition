45 void UpdateChecker::OnUpdateCheckCompleted() {
46  QNetworkReply *reply = qobject_cast<QNetworkReply *>(QObject::sender());
47  QByteArray bytes = reply->readAll();
48  int available_version = QString(bytes).toInt();
49  if (available_version > VERSION_CODE) {
50  emit UpdateAvailable();
51  }
