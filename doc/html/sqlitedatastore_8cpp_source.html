30 SqliteDataStore::SqliteDataStore(const std::string &filename) :
31  filename_(filename)
32 {
33  QDir dir(QDir::cleanPath((filename + "/..").c_str()));
34  if (!dir.exists())

52 std::string SqliteDataStore::Get(const std::string &key, const std::string &default_value) {
53  std::string query = "SELECT value FROM data WHERE key = ?";
54  sqlite3_stmt *stmt;
55  sqlite3_prepare(db_, query.c_str(), -1, &stmt, 0);
56  sqlite3_bind_text(stmt, 1, key.c_str(), -1, SQLITE_STATIC);
57  std::string result(default_value);
58  if (sqlite3_step(stmt) == SQLITE_ROW)
59  result = std::string(static_cast<const char*>(sqlite3_column_blob(stmt, 0)), sqlite3_column_bytes(stmt, 0));
60  sqlite3_finalize(stmt);
61  return result;
62 }

128 std::vector<CurrencyUpdate> SqliteDataStore::GetAllCurrency() {
129  std::string query = "SELECT timestamp, value FROM currency ORDER BY timestamp ASC";
130  sqlite3_stmt* stmt;
131  sqlite3_prepare(db_, query.c_str(), -1, &stmt, 0);
132  std::vector<CurrencyUpdate> result;
133  while (sqlite3_step(stmt) == SQLITE_ROW) {
134  CurrencyUpdate update = CurrencyUpdate();
135  update.timestamp = sqlite3_column_int64(stmt, 0);
136  update.value = std::string(reinterpret_cast<const char*>(sqlite3_column_text(stmt, 1)));
137  result.push_back(update);
138  }
139  sqlite3_finalize(stmt);
140  return result;
141 }
