50 std::string SqliteDataStore::Get(const std::string &key, const std::string &default_value) {
51  std::string query = "SELECT value FROM data WHERE key = ?";
52  sqlite3_stmt *stmt;
53  sqlite3_prepare(db_, query.c_str(), -1, &stmt, 0);
54  sqlite3_bind_text(stmt, 1, key.c_str(), -1, SQLITE_STATIC);
55  std::string result(default_value);
56  if (sqlite3_step(stmt) == SQLITE_ROW)
57  result = std::string(static_cast<const char*>(sqlite3_column_blob(stmt, 0)), sqlite3_column_bytes(stmt, 0));
58  sqlite3_finalize(stmt);
59  return result;
60 }

82 std::vector<CurrencyUpdate> SqliteDataStore::GetAllCurrency() {
83  std::string query = "SELECT timestamp, value FROM currency ORDER BY timestamp ASC";
84  sqlite3_stmt* stmt;
85  sqlite3_prepare(db_, query.c_str(), -1, &stmt, 0);
86  std::vector<CurrencyUpdate> result;
87  while (sqlite3_step(stmt) == SQLITE_ROW) {
88  CurrencyUpdate update = CurrencyUpdate();
89  update.timestamp = sqlite3_column_int64(stmt, 0);
90  update.value = std::string(reinterpret_cast<const char*>(sqlite3_column_text(stmt, 1)));
91  result.push_back(update);
92  }
93  sqlite3_finalize(stmt);
94  return result;
