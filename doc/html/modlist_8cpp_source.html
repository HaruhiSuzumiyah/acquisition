151 bool SumModGenerator::Match(const char *mod, double *output) {
152  bool found = false;
153  *output = 0.0;
154  for (auto &match : matches_) {
155  double result = 0.0;
156  if (Util::MatchMod(match.c_str(), mod, &result)) {
157  *output += result;
158  found = true;
159  }
160  }
161 
162  return found;
163 }

165 void SumModGenerator::Generate(const rapidjson::Value &json, ModTable *output) {
166  bool mod_present = false;
167  double sum = 0;
168  for (auto &type : { "implicitMods", "explicitMods", "craftedMods" }) {
169  if (!json.HasMember(type) || !json[type].IsArray())
170  continue;
171  for (auto &mod : json[type]) {
172  if (!mod.IsString())
173  continue;
174 
175  double result;
176  if (Match(mod.GetString(), &result)) {
177  sum += result;
178  mod_present = true;
179  }
180  }
181  }
182 
183  if (mod_present)
184  (*output)[name_] = sum;
185 }
