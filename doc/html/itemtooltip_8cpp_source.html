97 static std::string FormatProperty(const ItemProperty &prop) {
98  if (prop.display_mode == 3) {
99  QString format(prop.name.c_str());
100  for (auto &value : prop.values)
101  format = format.arg(ColorPropertyValue(value).c_str());
102  return format.toStdString();
103  }
104  std::string text = prop.name;
105  if (prop.values.size()) {
106  if (prop.name.size() > 0)
107  text += ": ";
108  bool first = true;
109  for (auto &value : prop.values) {
110  if (!first)
111  text += ", ";
112  first = false;
113  text += ColorPropertyValue(value);
114  }
115  }
116  return text;
117 }

169 static std::string GenerateItemInfo(const Item &item, const std::string &key, bool fancy) {
170  std::vector<std::string> sections;
171 
172  std::string properties_text = GenerateProperties(item);
173  if (properties_text.size() > 0)
174  sections.push_back(properties_text);
175 
176  std::string requirements_text = GenerateRequirements(item);
177  if (requirements_text.size() > 0)
178  sections.push_back(requirements_text);
179 
180  std::vector<std::string> mods = GenerateMods(item);
181  sections.insert(sections.end(), mods.begin(), mods.end());
182 
183  std::string unmet;
184  if (!item.identified())
185  unmet += "Unidentified";
186  if (item.corrupted())
187  unmet += (unmet.empty() ? "" : "<br>") + std::string("Corrupted");
188  if (!unmet.empty())
189  sections.push_back(ColorPropertyValue(ItemPropertyValue{ unmet, 2 }));
190 
191  std::string text;
192  bool first = true;
193  for (auto &s : sections) {
194  if (!first) {
195  text += "<br>";
196  if (fancy)
197  text += "<img src=':/tooltip/Separator" + key + ".png'><br>";
198  else
199  text += "<hr>";
200  }
201  first = false;
202  text += s;
203  }
204  if (!fancy)
205  text = ColorPropertyValue(ItemPropertyValue{ item.PrettyName(), 0 }) + "<hr>" + text;
206  return "<center>" + text + "</center>";
207 }

242 void GenerateItemHeaderSide(QLabel *itemHeader, bool leftNotRight, std::string header_path_prefix, bool singleline, Item::BASE_TYPES base) {
243  QImage header(QString::fromStdString(header_path_prefix + (leftNotRight ? "Left.png" : "Right.png")));
244  QSize header_size = singleline ? HEADER_SINGLELINE_SIZE : HEADER_DOUBLELINE_SIZE;
245  header = header.scaled(header_size);
246 
247  QPixmap header_pixmap(header_size);
248  header_pixmap.fill(Qt::transparent);
249  QPainter header_painter(&header_pixmap);
250  header_painter.drawImage(0, 0, header);
251 
252  if (base == Item::BASE_SHAPER || base == Item::BASE_ELDER) {
253  QImage overlay_image = base == Item::BASE_SHAPER ? shaper_icon : elder_icon;
254  overlay_image = overlay_image.scaled(HEADER_OVERLAY_SIZE);
255  int overlay_x = singleline ? (leftNotRight ? 2: 1) : (leftNotRight ? 2: 15);
256  int overlay_y = (int) ( 0.5 * (header.height() - overlay_image.height()) );
257  header_painter.drawImage(overlay_x, overlay_y, overlay_image);
258  }
259 
260  itemHeader->setFixedSize(header_size);
261  itemHeader->setPixmap(header_pixmap);
262 }

272  UpdateMinimap(item, ui);
273 
274  bool singleline = item.name().empty();

301 QPixmap GenerateItemSockets(const int width, const int height, const std::vector<ItemSocket> &sockets) {
302  QPixmap pixmap(width * PIXELS_PER_SLOT, height * PIXELS_PER_SLOT); // This will ensure we have enough room to draw the slots
303  pixmap.fill(Qt::transparent);
304  QPainter painter(&pixmap);
305 
306  int socket_rows = 0;
307  int socket_columns = 0;
308  ItemSocket prev = { 255, '-' };
309  size_t i = 0;
310 
311  if (sockets.size() == 0) {
312  // Do nothing
313  } else if (sockets.size() == 1) {
314  auto &socket = sockets.front();
315  QImage socket_image(":/sockets/" + QString(socket.attr) + ".png");
316  painter.drawImage(0, PIXELS_PER_SLOT * i, socket_image);
317  socket_rows = 1;
318  socket_columns = 1;
319  } else {
320  for (auto &socket : sockets) {
321  bool link = socket.group == prev.group;
322  QImage socket_image(":/sockets/" + QString(socket.attr) + ".png");
323  if (width == 1) {
324  painter.drawImage(0, PIXELS_PER_SLOT * i, socket_image);
325  if (link)
326  painter.drawImage(16, PIXELS_PER_SLOT * i - 19, link_v);
327  socket_columns = 1;
328  socket_rows = i + 1;
329  } else /* w == 2 */ {
330  int row = i / 2;
331  int column = i % 2;
332  if (row % 2 == 1)
333  column = 1 - column;
334  socket_columns = qMax(column + 1, socket_columns);
335  socket_rows = qMax(row + 1, socket_rows);
336  painter.drawImage(PIXELS_PER_SLOT * column, PIXELS_PER_SLOT * row, socket_image);
337  if (link) {
338  switch(i){
339  case 1:
340  case 3:
341  case 5:
342  // horizontal link
343  painter.drawImage(
344  PIXELS_PER_SLOT - LINKH_WIDTH / 2,
345  row * PIXELS_PER_SLOT + PIXELS_PER_SLOT / 2 - LINKH_HEIGHT / 2,
346  link_h
347  );
348  break;
349  case 2:
350  painter.drawImage(
351  PIXELS_PER_SLOT * 1.5 - LINKV_WIDTH / 2,
352  row * PIXELS_PER_SLOT - LINKV_HEIGHT / 2,
353  link_v
354  );
355  break;
356  case 4:
357  painter.drawImage(
358  PIXELS_PER_SLOT / 2 - LINKV_WIDTH / 2,
359  row * PIXELS_PER_SLOT - LINKV_HEIGHT / 2,
360  link_v
361  );
362  break;
363  default:
364  QLOG_ERROR() << "No idea how to draw link for socket " << i;
365  break;
366  }
367  }
368  }
369 
370  prev = socket;
371  ++i;
372  }
373  }
374 
375  return pixmap.copy(0, 0, PIXELS_PER_SLOT * socket_columns,
376  PIXELS_PER_SLOT * socket_rows);
377 }

379 QPixmap GenerateItemIcon(const Item &item, const QImage &image) {
380  const int height = item.h();
381  const int width = item.w();
382 
383  QPixmap layered(image.width(), image.height());
384  layered.fill(Qt::transparent);
385  QPainter layered_painter(&layered);
386 
387  if (item.shaper() || item.elder()){
388  // Assumes width <= 2
389  const QImage *background_image = nullptr;
390  if (item.shaper()) {
391  switch(height) {
392  case 1:
393  background_image = width == 1 ? &shaper_1x1 : &shaper_2x1;
394  break;
395  case 2:
396  background_image = width == 1 ? nullptr : &shaper_2x2;
397  break;
398  case 3:
399  background_image = width == 1 ? &shaper_1x3 : &shaper_2x3;
400  break;
401  case 4:
402  background_image = width == 1 ? &shaper_1x4 : &shaper_2x4;
403  break;
404  default:
405  break;
406  }
407  } else { // Elder
408  switch(height) {
409  case 1:
410  background_image = width == 1 ? &elder_1x1 : &elder_2x1;
411  break;
412  case 2:
413  background_image = width == 1 ? nullptr : &elder_2x2;
414  break;
415  case 3:
416  background_image = width == 1 ? &elder_1x3 : &elder_2x3;
417  break;
418  case 4:
419  background_image = width == 1 ? &elder_1x4 : &elder_2x4;
420  break;
421  default:
422  break;
423  }
424  }
425  if (background_image) {
426  layered_painter.drawImage(0, 0, *background_image);
427  } else {
428  QLOG_ERROR() << "Problem drawing background for " << item.PrettyName().c_str();
429  }
430  }
431 
432  layered_painter.drawImage(0, 0, image);
433 
434  if (item.text_sockets().size() > 0) {
435  QPixmap sockets = GenerateItemSockets(width, height, item.text_sockets());
436 
437  layered_painter.drawPixmap((int)(0.5*(image.width() - sockets.width())),
438  (int)(0.5*(image.height() - sockets.height())), sockets); // Center sockets on overall image
439  }
440 
441  return layered;
442 }
