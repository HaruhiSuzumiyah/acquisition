101 static std::string FormatProperty(const ItemProperty &prop) {
102  if (prop.display_mode == 3) {
103  QString format(prop.name.c_str());
104  for (auto &value : prop.values)
105  format = format.arg(ColorPropertyValue(value).c_str());
106  return format.toStdString();
107  }
108  std::string text = prop.name;
109  if (prop.values.size()) {
110  if (prop.name.size() > 0)
111  text += ": ";
112  bool first = true;
113  for (auto &value : prop.values) {
114  if (!first)
115  text += ", ";
116  first = false;
117  text += ColorPropertyValue(value);
118  }
119  }
120  return text;
121 }

173 static std::string GenerateItemInfo(const Item &item, const std::string &key, bool fancy) {
174  std::vector<std::string> sections;
175 
176  std::string properties_text = GenerateProperties(item);
177  if (properties_text.size() > 0)
178  sections.push_back(properties_text);
179 
180  std::string requirements_text = GenerateRequirements(item);
181  if (requirements_text.size() > 0)
182  sections.push_back(requirements_text);
183 
184  std::vector<std::string> mods = GenerateMods(item);
185  sections.insert(sections.end(), mods.begin(), mods.end());
186 
187  std::string unmet;
188  if (!item.identified())
189  unmet += "Unidentified";
190  if (item.corrupted())
191  unmet += (unmet.empty() ? "" : "<br>") + std::string("Corrupted");
192  if (!unmet.empty())
193  sections.push_back(ColorPropertyValue(ItemPropertyValue{ unmet, 2 }));
194 
195  std::string text;
196  bool first = true;
197  for (auto &s : sections) {
198  if (!first) {
199  text += "<br>";
200  if (fancy)
201  text += "<img src=':/tooltip/Separator" + key + ".png'><br>";
202  else
203  text += "<hr>";
204  }
205  first = false;
206  text += s;
207  }
208  if (!fancy)
209  text = ColorPropertyValue(ItemPropertyValue{ item.PrettyName(), 0 }) + "<hr>" + text;
210  return "<center>" + text + "</center>";
211 }

278  case Item::NONE:
279  break;

329 QPixmap GenerateItemSockets(const int width, const int height, const std::vector<ItemSocket> &sockets) {
330  QPixmap pixmap(width * PIXELS_PER_SLOT, height * PIXELS_PER_SLOT); // This will ensure we have enough room to draw the slots
331  pixmap.fill(Qt::transparent);
332  QPainter painter(&pixmap);
333 
334  int socket_rows = 0;
335  int socket_columns = 0;
336  ItemSocket prev = { 255, '-' };
337  size_t i = 0;
338 
339  if (sockets.size() == 0) {
340  // Do nothing
341  } else if (sockets.size() == 1) {
342  auto &socket = sockets.front();
343  QImage socket_image(":/sockets/" + QString(socket.attr) + ".png");
344  painter.drawImage(0, PIXELS_PER_SLOT * i, socket_image);
345  socket_rows = 1;
346  socket_columns = 1;
347  } else {
348  for (auto &socket : sockets) {
349  bool link = socket.group == prev.group;
350  QImage socket_image(":/sockets/" + QString(socket.attr) + ".png");
351  if (width == 1) {
352  painter.drawImage(0, PIXELS_PER_SLOT * i, socket_image);
353  if (link)
354  painter.drawImage(16, PIXELS_PER_SLOT * i - 19, link_v);
355  socket_columns = 1;
356  socket_rows = i + 1;
357  } else /* w == 2 */ {
358  int row = i / 2;
359  int column = i % 2;
360  if (row % 2 == 1)
361  column = 1 - column;
362  socket_columns = qMax(column + 1, socket_columns);
363  socket_rows = qMax(row + 1, socket_rows);
364  painter.drawImage(PIXELS_PER_SLOT * column, PIXELS_PER_SLOT * row, socket_image);
365  if (link) {
366  switch(i){
367  case 1:
368  case 3:
369  case 5:
370  // horizontal link
371  painter.drawImage(
372  PIXELS_PER_SLOT - LINKH_WIDTH / 2,
373  row * PIXELS_PER_SLOT + PIXELS_PER_SLOT / 2 - LINKH_HEIGHT / 2,
374  link_h
375  );
376  break;
377  case 2:
378  painter.drawImage(
379  PIXELS_PER_SLOT * 1.5 - LINKV_WIDTH / 2,
380  row * PIXELS_PER_SLOT - LINKV_HEIGHT / 2,
381  link_v
382  );
383  break;
384  case 4:
385  painter.drawImage(
386  PIXELS_PER_SLOT / 2 - LINKV_WIDTH / 2,
387  row * PIXELS_PER_SLOT - LINKV_HEIGHT / 2,
388  link_v
389  );
390  break;
391  default:
392  QLOG_ERROR() << "No idea how to draw link for socket " << i;
393  break;
394  }
395  }
396  }
397 
398  prev = socket;
399  ++i;
400  }
401  }
402 
403  return pixmap.copy(0, 0, PIXELS_PER_SLOT * socket_columns,
404  PIXELS_PER_SLOT * socket_rows);
405 }

407 QPixmap GenerateItemIcon(const Item &item, const QImage &image) {
408  const int height = item.h();
409  const int width = item.w();
410 
411  QPixmap layered(image.width(), image.height());
412  layered.fill(Qt::transparent);
413  QPainter layered_painter(&layered);
414 
415  if (item.hasBaseType(Item::BASE_SHAPER) || item.hasBaseType(Item::BASE_ELDER)){
416  // Assumes width <= 2
417  const QImage *background_image = nullptr;
418  if (item.hasBaseType(Item::BASE_ELDER)) {
419  switch(height) {
420  case 1:
421  background_image = width == 1 ? &shaper_1x1 : &shaper_2x1;
422  break;
423  case 2:
424  background_image = width == 1 ? nullptr : &shaper_2x2;
425  break;
426  case 3:
427  background_image = width == 1 ? &shaper_1x3 : &shaper_2x3;
428  break;
429  case 4:
430  background_image = width == 1 ? &shaper_1x4 : &shaper_2x4;
431  break;
432  default:
433  break;
434  }
435  } else { // Elder
436  switch(height) {
437  case 1:
438  background_image = width == 1 ? &elder_1x1 : &elder_2x1;
439  break;
440  case 2:
441  background_image = width == 1 ? nullptr : &elder_2x2;
442  break;
443  case 3:
444  background_image = width == 1 ? &elder_1x3 : &elder_2x3;
445  break;
446  case 4:
447  background_image = width == 1 ? &elder_1x4 : &elder_2x4;
448  break;
449  default:
450  break;
451  }
452  }
453  if (background_image) {
454  layered_painter.drawImage(0, 0, *background_image);
455  } else {
456  QLOG_ERROR() << "Problem drawing background for " << item.PrettyName().c_str();
457  }
458  }
459 
460  layered_painter.drawImage(0, 0, image);
461 
462  if (item.text_sockets().size() > 0) {
463  QPixmap sockets = GenerateItemSockets(width, height, item.text_sockets());
464 
465  layered_painter.drawPixmap((int)(0.5*(image.width() - sockets.width())),
466  (int)(0.5*(image.height() - sockets.height())), sockets); // Center sockets on overall image
467  }
468 
469  return layered;
470 }
