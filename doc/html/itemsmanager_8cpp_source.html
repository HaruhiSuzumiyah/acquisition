69 void ItemsManager::ApplyAutoTabBuyouts() {
70  // Can handle everything related to auto-tab pricing here.
71  // 1. First format we need to honor is ascendency pricing formats which is top priority and overrides other types
72  // 2. Second priority is to honor manual user pricing
73  // 3. Third priority it to apply pricing based on ideally user specified formats (doesn't exist yet)
74 
75  // Loop over all tabs, create buyout based on tab name which applies auto-pricing policies
76  auto &bo = app_.buyout_manager();
77  for (auto const &loc: bo_manager_.GetStashTabLocations()) {
78  auto tab_label = loc.get_tab_label();
79  Buyout buyout = bo.StringToBuyout(tab_label);
80  if (buyout.IsActive()) {
81  bo.SetTab(loc.GetUniqueHash(), buyout);
82  }
83  }
84 
85  // Need to compress tab buyouts here, as the tab names change we accumulate and save BO's
86  // for tabs that no longer exist I think.
87  bo.CompressTabBuyouts();
88 }

90 void ItemsManager::ApplyAutoItemBuyouts() {
91  // Loop over all items, check for note field with pricing and apply
92  auto &bo = app_.buyout_manager();
93  for (auto const& item: items_) {
94  auto const &note = item->note();
95  if (!note.empty()) {
96  Buyout buyout = bo.StringToBuyout(note);
97  // This line may look confusing, buyout returns an active buyout if game
98  // pricing was found or a default buyout (inherit) if it was not.
99  // If there is a currently valid note we want to apply OR if
100  // old note no longer is valid (so basically clear pricing)
101  if (buyout.IsActive() || bo.Get(*item).IsGameSet()) {
102  bo.Set(*item, buyout);
103  }
104  }
105  }
106 
107  // Commenting this out for robustness (iss381) to make it as unlikely as possible that users
108  // pricing data will be removed. Side effect is that stale pricing data will pile up and
109  // could be applied to future items with the same hash (which includes tab name).
110  // bo.CompressItemBuyouts(items_);
111 }

113 void ItemsManager::PropagateTabBuyouts() {
114  auto &bo = app_.buyout_manager();
115  bo.ClearRefreshLocks();
116  for (auto &item_ptr : items_) {
117  Item &item = *item_ptr;
118  std::string hash = item.location().GetUniqueHash();
119  auto item_bo = bo.Get(item);
120  auto tab_bo = bo.GetTab(hash);
121 
122  if (item_bo.IsInherited()) {
123  if (tab_bo.IsActive()) {
124  // Any propagation from tab price to item price should include this bit set
125  tab_bo.inherited = true;
126  tab_bo.last_update = QDateTime::currentDateTime();
127  bo.Set(item, tab_bo);
128  } else {
129  // This effectively 'clears' buyout by setting back to 'inherit' state.
130  bo.Set(item, Buyout());
131  }
132  }
133 
134  // If any savable bo's are set on an item or the tab then lock
135  // the refresh state.
136  if (bo.Get(item).RequiresRefresh() || tab_bo.RequiresRefresh()) {
137  bo.SetRefreshLocked(item.location());
138  }
139  }
140 }
