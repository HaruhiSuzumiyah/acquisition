91 std::string Util::GetCsrfToken(const std::string &page, const std::string &name) {
92  std::string needle = "name=\"" + name + "\" value=\"";
93  auto pos = page.find(needle);
94  if (pos == std::string::npos)
95  return "";
96  auto start = pos + needle.size(); // Right after the opening "
97  auto end = page.find("\"", start); // Closing "
98  if (end == std::string::npos)
99  return "";
100  return page.substr(start, end - start);
101 }

152 bool Util::MatchMod(const char *match, const char *mod, double *output) {
153  double result = 0.0;
154  auto pmatch = match;
155  auto pmod = mod;
156  int cnt = 0;
157 
158  while (*pmatch && *pmod) {
159  if (*pmatch == '#') {
160  ++cnt;
161  auto prev = pmod;
162  while ((*pmod >= '0' && *pmod <= '9') || *pmod == '.')
163  ++pmod;
164  result += std::strtod(prev, NULL);
165  ++pmatch;
166  } else if (*pmatch == *pmod) {
167  ++pmatch;
168  ++pmod;
169  } else {
170  return false;
171  }
172  }
173  *output = result / cnt;
174  return !*pmatch && !*pmod;
175 }

183 std::string Util::TimeAgoInWords(const QDateTime buyout_time){
184  QDateTime current_date = QDateTime::currentDateTime();
185  qint64 secs = buyout_time.secsTo(current_date);
186  qint64 days = secs / 60 / 60 / 24;
187  qint64 hours = (secs / 60 / 60) % 24;
188  qint64 minutes = (secs / 60) % 60;
189 
190  // YEARS
191  if (days > 365){
192  int years = (days / 365);
193  if (days % 365 != 0)
194  years++;
195  return QString("%1 %2 ago").arg(years).arg(years == 1 ? "year" : "years").toStdString();
196  }
197  // MONTHS
198  if (days > 30){
199  int months = (days / 365);
200  if (days % 30 != 0)
201  months++;
202  return QString("%1 %2 ago").arg(months).arg(months == 1 ? "month" : "months").toStdString();
203  // DAYS
204  }else if (days > 0){
205  return QString("%1 %2 ago").arg(days).arg(days == 1 ? "day" : "days").toStdString();
206  // HOURS
207  }else if (hours > 0){
208  return QString("%1 %2 ago").arg(hours).arg(hours == 1 ? "hour" : "hours").toStdString();
209  //MINUTES
210  }else if (minutes > 0){
211  return QString("%1 %2 ago").arg(minutes).arg(minutes == 1 ? "minute" : "minutes").toStdString();
212  // SECONDS
213  }else if (secs > 5){
214  return QString("%1 %2 ago").arg(secs).arg("seconds").toStdString();
215  }else if (secs < 5){
216  return QString("just now").toStdString();
217  }else{
218  return "";
219  }
220 }

228 QColor Util::recommendedForegroundTextColor(const QColor& backgroundColor){
229  float R = (float) backgroundColor.red() / 255.0f;
230  float G = (float) backgroundColor.green() / 255.0f;
231  float B = (float) backgroundColor.blue() / 255.0f;
232 
233  const float gamma = 2.2;
