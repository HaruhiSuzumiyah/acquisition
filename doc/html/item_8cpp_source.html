59 static std::string item_unique_properties(const rapidjson::Value &json, const std::string &name) {
60  const char *name_p = name.c_str();
61  if (!json.HasMember(name_p))
62  return "";
63  std::string result;
64  for (auto &prop : json[name_p]) {
65  result += std::string(prop["name"].GetString()) + "~";
66  for (auto &value : prop["values"])
67  result += std::string(value[0].GetString()) + "~";
68  }
69  return result;
70 }

294 void Item::CalculateCategories(const rapidjson::Value &json) {
295  // Derive item type 'category' hierarchy from icon path.
296  std::smatch sm;
297  if (std::regex_search(icon_, sm, std::regex("Art/.*?/(.*)/"))) {
298  std::string match = sm.str(1);
299  boost::split(category_vector_,match,boost::is_any_of("/"));
300  //Compress terms with redundant identifiers
301  //Weapons.OneHandWeapons.OneHandMaces -> Weapons.OneHand.Maces
302  size_t min = std::min(category_vector_.size(), replace_map_.size());
303  for (size_t i = 0; i < min; i++) {
304  auto it = replace_map_[i].find(category_vector_[i]);
305  if (it != replace_map_[i].end())
306  category_vector_[i] = it->second;
307  }
308  } else if (std::regex_search(icon_, sm, std::regex("/gen/image/.*?/Item.png"))) {
309  // Flask images are dynamically generated by GGG to reflect current charge status, rather than live under /Art/2DItems/
310  category_vector_.push_back("Flasks");
311  } else {
312  category_vector_.push_back("Unknown");
313  }
314 
315  if (category_vector_[0] == "Jewels") {
316  if (std::regex_search(icon_, sm, std::regex("/Jewels/.+?Eye.png"))) {
317  category_vector_.push_back("Abyss");
318  }
319  } else if (json.HasMember("prophecyText") && json["prophecyText"].IsString()) {
320  // Relocate Prophecies out from Currencies
321  category_vector_[0] = "Prophecies";
322  } else if (category_vector_[0] == "Maps") {
323  if (std::regex_search(icon_, sm, std::regex("Maps/(?:Uber)?Vaal(?:[[:digit:]]){2}.png"))) {
324  // Check for digits immediately after Vaal, to avoid matching on VaalCity (AKA Ancient City), VaalTemple
325  category_vector_.push_back("Atziri Fragments");
326  // What about Prophecy key framents, atlas guardian drop fragments, scarabs, divine vessels? For now most end up under Misc (GGG classed scarabs as currency in their icon path)
327  } else if (std::regex_search(icon_, sm, std::regex("Maps/(?:.*)Shard.png"))) {
328  // Check for Legion splinters, recategorise as currency. What about Emblems, need an example to test.
329  category_vector_[0] = "Currency";
330  category_vector_.push_back("Legion");
331  } else {
332  if (category_vector_.size() > 1) {
333  if (category_vector_[1] == "Atlas2Maps") {
334  // Try to give newer maps a more useful category name
335  category_vector_[1] = "3";
336  if (category_vector_.size() > 2 ) {
337  // More accurately 3.1+
338  if (category_vector_[2] == "New") {
339  category_vector_[2] = "1";
340  // Calculate release version from ?mn= parameter
341  if (std::regex_search(icon_, sm, std::regex("&mn=([[:digit:]])+"))) {
342  int mn = std::stoi(sm[1], nullptr, 10);
343  if (mn > 1) {
344  // mn became 2 for 3.5.0
345  category_vector_[2] = std::to_string(mn + 3);
346  }
347  }
348  }
349  } else {
350  category_vector_.push_back("0"); // Were any added to Atlas2Maps but not New?
351  }
352  }
353  } else {
354  if (icon_.find("/Art/2DItems/Maps/Map") != std::string::npos) {
355  // Doesn't find all because of some maps like FairgravesMap01.png, olmec.png, etc. We'll pick them out of Misc later
356  category_vector_.push_back("Original");
357  } else {
358  category_vector_.push_back("Misc");
359 
360  // Check that these aren't just badly named actual maps
361  if (json.HasMember("properties") && json["properties"].IsArray()) {
362  for (auto &prop : json["properties"]) {
363  if (!prop.HasMember("name") || !prop["name"].IsString() || !prop.HasMember("values") || !prop["values"].IsArray())
364  continue;
365  std::string name = prop["name"].GetString();
366  if (name == "Map Tier") {
367  category_vector_.pop_back();
368  category_vector_.push_back("Older Uniques");
369  break;
370  }
371  }
372  }
373  }
374  }
375  }
376  }
377 
378  if (category_vector_.size() > 1 && category_vector_[1] == "Scarabs") {
379  category_vector_[0] = "Maps";
380  // Technically scarabs are fragments. Better to leave Fragments as just Atziri ones, give scarabs their own subcategory?
381  }
382 
383  category_ = boost::join(category_vector_, ".");
384  boost::to_lower(category_);
385 
386 }

425 void Item::CalculateHash(const rapidjson::Value &json) {
426  std::string unique_new = name_ + "~" + typeLine_ + "~";
427  // GGG removed the <<set>> things in patch 3.4.3e but our hashes all include them, oops
428  std::string unique_old = "<<set:MS>><<set:M>><<set:S>>" + unique_new;
429 
430  std::string unique_common;
431 
432  if (json.HasMember("explicitMods") && json["explicitMods"].IsArray())
433  for (auto &mod : json["explicitMods"])
434  if (mod.IsString())
435  unique_common += std::string(mod.GetString()) + "~";
436 
437  if (json.HasMember("implicitMods") && json["implicitMods"].IsArray())
438  for (auto &mod : json["implicitMods"])
439  if (mod.IsString())
440  unique_common += std::string(mod.GetString()) + "~";
441 
442  unique_common += item_unique_properties(json, "properties") + "~";
443  unique_common += item_unique_properties(json, "additionalProperties") + "~";
444 
445  if (json.HasMember("sockets") && json["sockets"].IsArray())
446  for (auto &socket : json["sockets"]) {
447  if (!socket.HasMember("group") || !socket.HasMember("attr") || !socket["group"].IsInt() || !socket["attr"].IsString())
448  continue;
449  unique_common += std::to_string(socket["group"].GetInt()) + "~" + socket["attr"].GetString() + "~";
450  }
451 
452  unique_common += "~" + location_.GetUniqueHash();
453 
454  unique_old += unique_common;
455  unique_new += unique_common;
456 
457  old_hash_ = Util::Md5(unique_old);
458  hash_ = Util::Md5(unique_new);
459 }

476 std::string Item::POBformat() const {
477  std::stringstream PoBText;
478  PoBText << name();
479  PoBText << "\n" << typeLine();
480 
481  // Could use uid_ for "Unique ID:", if it'd help PoB avoid duplicate imports later via stash API?
482 
483  auto &sockets = text_sockets();
484  if (sockets.size() > 0) {
485  PoBText << "\nSockets: ";
486  ItemSocket prev = { 255, '-' };
487  size_t i = 0;
488  for (auto &socket : sockets) {
489  bool link = socket.group == prev.group;
490  if (i > 0) {
491  PoBText << (link ? "-" : " ");
492  }
493  switch (socket.attr) {
494  case 'S':
495  PoBText << "R";
496  break;
497  case 'D':
498  PoBText << "G";
499  break;
500  case 'I':
501  PoBText << "B";
502  break;
503  case 'G':
504  PoBText << "W";
505  break;
506  default:
507  PoBText << socket.attr;
508  break;
509  }
510  prev = socket;
511  ++i;
512  }
513  }
514 
515  auto &mods = text_mods();
516 
517  auto implicitMods = mods.at("implicitMods");
518  auto enchantMods = mods.at("enchantMods");
519  PoBText << "\nImplicits: " << (implicitMods.size() + enchantMods.size());
520  for (const auto &mod : enchantMods) {
521  PoBText << "\n{crafted}" << mod;
522  }
523  for (const auto &mod : implicitMods) {
524  PoBText << "\n"<< mod;
525  }
526 
527  auto explicitMods = mods.at("explicitMods");
528  auto craftedMods = mods.at("craftedMods");
529  if (!explicitMods.empty() || !craftedMods.empty()) {
530  for (const auto &mod : explicitMods) {
531  PoBText << "\n"<< mod;
532  }
533  for (const auto &mod : craftedMods) {
534  PoBText << "\n{crafted}" << mod;
535  }
536  }
537 
538  return PoBText.str();
539 }
