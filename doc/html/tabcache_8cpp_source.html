7 // TabCache
8 //
9 // This class is a 'work around' for path of exile API not using 'ETag' headers
10 // and supporting existing cache methods that QNetworkDiskCache would have
11 // otherwise supported directly. If they ever fix this we should be able to
12 // just remove this implementation and use QNetworkDiskCache directly.
13 //
14 // Currently the API sends headers that look like this:
15 //
16 // DEBUG 2016-01-16T13:04:10.054 "Server" "nginx/1.4.4"
17 // DEBUG 2016-01-16T13:04:10.054 "Date" "Sat, 16 Jan 2016 19:04:10 GMT"
18 // DEBUG 2016-01-16T13:04:10.054 "Content-Type" "application/json"
19 // DEBUG 2016-01-16T13:04:10.054 "Expires" "Thu, 19 Nov 1981 08:52:00 GMT"
20 // DEBUG 2016-01-16T13:04:10.055 "Cache-Control" "no-store, no-cache, must-revalidate, post-check=0, pre-check=0"
21 // DEBUG 2016-01-16T13:04:10.055 "Pragma" "no-cache"
22 // DEBUG 2016-01-16T13:04:10.055 "X-Frame-Options" "SAMEORIGIN"
23 //
24 // 'no-cache' - actually allows client to cache requests, but *requests*
25 // revalidation to use cached data (using ETag or Last Modified headers).
26 // clients can choose to ignore revalidation request and use possibly
27 // stale content.
28 //
29 // 'must-revalidate' - *requires* revalidation of cached content. If no
30 // revalidation method exists don't use cached data.
31 //
32 // 'no-store' - *requires* client to not store content to disk. This is the what
33 // really disables caching.
34 //
35 // So the plan is to basically just to ignore the Cache-Control and Pragma headers.
36 // This is achieved by overridding QNetworkDiskCache 'prepare' method so it
37 // effectively doesn't seem them.
38 //
39 // A helper method is needed however to create our network requests properly such
40 // such that the requests always prefer hitting in the cache.
41 //
42 // The method is:
43 //
44 // Request(url, <flags>);
45 //
46 // Where <flags> can be used to force an eviction, causing a reload
47 // Request(url, TabCache::Refresh);
48 //
49 // If 'Refresh' is not specified we're guaranteed to hit in cache if entry exists
50 // and fetch otherwise.
51 //
52 // So, to minimize overall API requests it will become a matter of minimizing requests
53 // that ask for a refresh. From the point of view of the rest of the application it will
54 // always appear as if we request all the tabs.

61 QNetworkRequest TabCache::Request(const QUrl &url, Flags flags) {
62  QNetworkRequest request{url};
63  bool evicted = false;
64 
65  if (flags.testFlag(Refresh)) {
66  remove(url);
67  evicted = true;
68  }
69 
70  // At this point we've evicted any request that should be refreshed, so we always
71  // tell the 'real' request to prefer but not require the entry be in the cache.
72  // If it is not in the cache it will be fetched from the network regardless.
73  request.setAttribute(QNetworkRequest::CacheSaveControlAttribute, QNetworkRequest::PreferCache);
74  QLOG_DEBUG() << "Evicted:" << evicted << ":" << url.toDisplayString();
75 
76  return request;
77 }

87  local.setSaveToDisk(true);

91  local.setExpirationDate(QDateTime().currentDateTime().addDays(kCacheExpireInDays));
92 
93  QNetworkCacheMetaData::RawHeaderList headers;

95  for (auto &header: local.rawHeaders()) {
96  // Modify Cache-Control headers - basically need to drop 'no-store'
97  // as we want to store to cache and 'must-revalidate' as we don't have
98  // ETag or last modified headers available to attempt re-validation. To
99  // be on the safe side though just drop Cache-Control and Pragma headers.
100

102  if (header.first == "Pragma") continue;

105  }
