37 /*
38  * Objects of subclasses of this class do the following:
39  * 1) FromForm: provided with a FilterData fill it with data from form
40  * 2) ToForm: provided with a FilterData fill form with data from it
41  * 3) Matches: check if an item matches the filter provided with FilterData
42  */
43 class Filter {
44 public:
45  virtual void FromForm(FilterData *data) = 0;
46  virtual void ToForm(FilterData *data) = 0;
47  virtual void ResetForm() = 0;
48  virtual bool Matches(const std::shared_ptr<Item> &item, FilterData *data) = 0;
49  virtual ~Filter() {};
50  std::unique_ptr<FilterData> CreateData();
51 };
