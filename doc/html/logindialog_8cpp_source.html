170 void LoginDialog::OnLeaguesRequestFinished() {
171  SelfDestructingReply reply(qobject_cast<QNetworkReply *>(QObject::sender()));
172  QByteArray bytes = reply->readAll();
173 
174  if (reply->error())
175  return LeaguesApiError(reply->errorString(), bytes);
176 
177  rapidjson::Document doc;
178  doc.Parse(bytes.constData());
179 
180  if (doc.HasParseError() || !doc.IsArray())
181  return LeaguesApiError("Failed to parse the document", bytes);
182 
183  ui->leagueComboBox->clear();
184  for (auto &league : doc) {
185  if (!league.IsObject())
186  return LeaguesApiError("Object expected", bytes);
187  if (!league.HasMember("id"))
188  return LeaguesApiError("Missing league 'id'", bytes);
189  if (!league["id"].IsString())
190  return LeaguesApiError("String expected", bytes);
191  ui->leagueComboBox->addItem(league["id"].GetString());
192  }
193  ui->leagueComboBox->setEnabled(true);
194 
195  if (saved_league_.size() > 0)
196  ui->leagueComboBox->setCurrentText(saved_league_);
197 }

205 void LoginDialog::OnLoginPageFinished() {
206  QNetworkReply *reply = qobject_cast<QNetworkReply *>(QObject::sender());
207  if (reply->error()) {
208  DisplayError("Network error: " + reply->errorString() + "\nTry using another login method.");
209  return;
210  }
211 
212  QByteArray bytes = reply->readAll();
213  std::string page(bytes.constData(), bytes.size());
214  std::string hash = Util::GetCsrfToken(page, "hash");
215  if (hash.empty()) {
216  DisplayError("Failed to log in (can't extract form hash from page)");
217  return;
218  }
219 
220  QUrlQuery query;
221  query.addQueryItem("login_email", EncodeSpecialCharacters(ui->emailLineEdit->text()));
222  query.addQueryItem("login_password", EncodeSpecialCharacters(ui->passwordLineEdit->text()));
223  query.addQueryItem("hash", QString(hash.c_str()));
224  query.addQueryItem("login", "Login");
225  query.addQueryItem("remember_me", "1");
226 
227  QUrl url(POE_LOGIN_URL);
228  QByteArray data(query.query().toUtf8());
229  QNetworkRequest request(url);
230  request.setHeader(QNetworkRequest::ContentTypeHeader, "application/x-www-form-urlencoded");
231  QNetworkReply *logged_in = login_manager_->post(request, data);
232  connect(logged_in, SIGNAL(finished()), this, SLOT(OnLoggedIn()));
233 }

290 void LoginDialog::OnMainPageFinished() {
291  QNetworkReply *reply = qobject_cast<QNetworkReply *>(QObject::sender());
292  QString html(reply->readAll());
293  QRegExp regexp("/account/view-profile/(.*)\"");
294  regexp.setMinimal(true);
295  int pos = regexp.indexIn(html);
296  if (pos == -1) {
297  DisplayError("Failed to find account name.");
298  return;
299  }
300  QString account = regexp.cap(1);
301  QLOG_DEBUG() << "Logged in as:" << account;
302 
303  std::string league(ui->leagueComboBox->currentText().toStdString());
304  app_->InitLogin(std::move(login_manager_), league, account.toStdString());
305  mw = new MainWindow(std::move(app_));
306  mw->setWindowTitle(QString("Acquisition - %1").arg(league.c_str()));
307  mw->show();
308  close();

332 void LoginDialog::SaveSettings() {
333  QSettings settings(settings_path_.c_str(), QSettings::IniFormat);
334  if (ui->rembmeCheckBox->isChecked()) {
335  settings.setValue("session_id", session_id_);
336  settings.setValue("league", ui->leagueComboBox->currentText());
337  } else {
338  settings.setValue("session_id", "");
339  settings.setValue("league", "");
340  }
341  settings.setValue("remember_me_checked", ui->rembmeCheckBox->isChecked() && !session_id_.isEmpty());
342  settings.setValue("use_system_proxy_checked", ui->proxyCheckBox->isChecked());
343 
344  // Clear any saved cookies (steamMachineAuth in particular) if user doesn't want
345  // us to remember login info.
346  //
347  // For details see: https://dev.doctormckay.com/topic/365-cookies/
348  //
349  // By default WebEngine stores cookies like a normal browser and steamMachineAuth
350  // cookie does not naturally expire. Without this cookie the user will be required to
351  // re-verify access through e-mail.
352 #ifndef NO_WEBENGINE
353  if (!settings.value("remember_me_checked").toBool()) {
354  QWebEngineProfile::defaultProfile()->cookieStore()->deleteAllCookies();
355  }
356 #endif
357 }
