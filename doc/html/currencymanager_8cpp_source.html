150 std::string CurrencyManager::Serialize(const std::vector<std::shared_ptr<CurrencyItem>> &currencies) {
151  rapidjson::Document doc;
152  doc.SetObject();
153  auto &alloc = doc.GetAllocator();
154  for (auto &curr : currencies) {
155  rapidjson::Value item(rapidjson::kObjectType);
156  item.AddMember("count", curr->count, alloc);
157  item.AddMember("chaos_ratio", curr->chaos.value1, alloc);
158  item.AddMember("exalt_ratio", curr->exalt.value1, alloc);
159  Util::RapidjsonAddConstString(&item, "currency", curr->currency.AsTag(), alloc);
160  rapidjson::Value name(curr->name.c_str(), alloc);
161  doc.AddMember(name, item, alloc);
162  }
163  return Util::RapidjsonSerialize(doc);
164 }

166 void CurrencyManager::Deserialize(const std::string &data, std::vector<std::shared_ptr<CurrencyItem> > *currencies) {
167  //Maybe clear something would be good
168  if (data.empty())
169  return;
170  rapidjson::Document doc;
171  if (doc.Parse(data.c_str()).HasParseError()) {
172  QLOG_ERROR() << "Error while parsing currency ratios.";
173  QLOG_ERROR() << rapidjson::GetParseError_En(doc.GetParseError());
174  return;
175  }
176  if (!doc.IsObject())
177  return;
178  for (auto itr = doc.MemberBegin(); itr != doc.MemberEnd(); ++itr) {
179  auto &object = itr->value;
180  Currency curr = Currency::FromTag(object["currency"].GetString());
181  for (auto &item : *currencies) {
182  if (item->currency == curr) {
183  item = std::make_shared<CurrencyItem>(object["count"].GetDouble(), curr,
184  object["chaos_ratio"].GetDouble(), object["exalt_ratio"].GetDouble());
185  }
186  }
187  //currencies->push_back(item);
188  }
189 }

195 void CurrencyManager::SaveCurrencyValue() {
196  std::string value = "";
197  // Useless to save if every count is 0.
198  bool empty = true;
199  value = std::to_string(TotalExaltedValue());
200  for (auto &currency : currencies_) {
201  if (currency->name != "")
202  value += ";" + std::to_string(currency->count);
203  if (currency->count != 0)
204  empty = false;
205  }
206  std::string old_value = data_.Get("currency_last_value", "");
207  if (value != old_value && !empty) {
208  CurrencyUpdate update = CurrencyUpdate();
209  update.timestamp = std::time(nullptr);
210  update.value = value;
211  data_.InsertCurrencyUpdate(update);
212  data_.Set("currency_last_value", value);
213  }
214 }

216 void CurrencyManager::ExportCurrency() {
217  std::string header_csv = "Date; Total value";
218  for (auto& item : currencies_) {
219  auto label = item->currency.AsString();
220  if (label != "")
221  header_csv += ";" + label;
222  }
223  std::vector<CurrencyUpdate> result = data_.GetAllCurrency();
224 
225  QString fileName = QFileDialog::getSaveFileName(this, tr("Save Export file"),
226  QDir::toNativeSeparators(QDir::homePath() + "/" + "acquisition_export_currency.csv"));
227  if (fileName.isEmpty())
228  return;
229  QFile file(QDir::toNativeSeparators(fileName));
230  if (file.open(QFile::WriteOnly | QFile::Text)) {
231  QTextStream out(&file);
232  out << header_csv.c_str() << "\n";
233  for (auto &update : result) {
234  char buf[4096];
235  std::time_t timestamp = update.timestamp;
236  std::strftime(buf, sizeof(buf), "%Y-%m-%d %H:%M", std::localtime(&timestamp));
237  out << buf << ";";
238  out << update.value.c_str() << "\n";
239  }
240  } else {
241  QLOG_WARN() << "CurrencyManager::ExportCurrency : couldn't open CSV export file ";
242  }
243 }
