323 void MainWindow::OnBuyoutChange() {
324  app_->shop().ExpireShopData();
325 
326  Buyout bo;
327  bo.type = Buyout::IndexAsBuyoutType(ui->buyoutTypeComboBox->currentIndex());
328  bo.currency = Currency::FromIndex(ui->buyoutCurrencyComboBox->currentIndex());
329  bo.value = ui->buyoutValueLineEdit->text().replace(',', ".").toDouble();
330  bo.last_update = QDateTime::currentDateTime();
331 
332  if (bo.IsPriced()) {
333  ui->buyoutCurrencyComboBox->setEnabled(true);
334  ui->buyoutValueLineEdit->setEnabled(true);
335  } else {
336  ui->buyoutCurrencyComboBox->setEnabled(false);
337  ui->buyoutValueLineEdit->setEnabled(false);
338  }
339 
340  if (!bo.IsValid())
341  return;
342 
343  // Don't assign a zero buyout if nothing is entered in the value textbox
344  if (ui->buyoutValueLineEdit->text().isEmpty() && bo.IsPriced())
345  return;
346 
347  BuyoutManager &bo_manager = app_->buyout_manager();
348  for (auto const &index: ui->treeView->selectionModel()->selectedIndexes()) {
349  auto const &tab = current_search_->GetTabLocation(index).GetUniqueHash();
350 
351  // Don't allow users to manually update locked tabs (game priced)
352  if (bo_manager.GetTab(tab).IsGameSet())
353  continue;
354  if (!index.parent().isValid()) {
355  bo_manager.SetTab(tab, bo);
356  } else {
357  auto &item = current_search_->bucket(index.parent().row())->item(index.row());
358  // Don't allow users to manually update locked items (game priced per item in note section)
359  if (bo_manager.Get(*item).IsGameSet())
360  continue;
361  bo_manager.Set(*item, bo);
362  }
363  }
364  app_->items_manager().PropagateTabBuyouts();
365  // refresh treeView to immediately reflect price changes
366  ui->treeView->model()->layoutChanged();
367  ResizeTreeColumns();
368 }

370 void MainWindow::OnStatusUpdate(const CurrentStatusUpdate &status) {
371  QString title;
372  bool need_progress = false;
373  switch (status.state) {
374  case ProgramState::CharactersReceived:
375  if (status.total == 0) {
376  refresh_button_.setText("No characters detected yet in this league, click to refresh");
377  refresh_button_.show();
378  } else {
379  refresh_button_.hide();
380  }
381  break;
382  case ProgramState::ItemsReceive:
383  case ProgramState::ItemsPaused:
384  title = QString("Receiving stash data, %1/%2 [%3 from cache]").arg(status.progress).arg(status.total).arg(status.cached);
385  if (status.state == ProgramState::ItemsPaused)
386  title += " (throttled, sleeping 60 seconds)";
387  need_progress = true;
388  break;
389  case ProgramState::ItemsCompleted:
390  title = QString("Received %1 tabs [%2 from cache]").arg(status.total).arg(status.cached);
391  QLOG_INFO() << title;
392  break;
393  case ProgramState::ShopSubmitting:
394  title = QString("Sending your shops to the forum, %1/%2").arg(status.progress).arg(status.total);
395  need_progress = true;
396  break;
397  case ProgramState::ShopCompleted:
398  title = QString("Shop threads updated");
399  QLOG_INFO() << title;
400  break;
401  case ProgramState::UpdateCancelled:
402  title = QString("Shop updated cancelled due to tab movement or rename mid-update");
403  break;
404  default:
405  title = "Unknown";
406  }
407 
408  status_bar_label_->setText(title);
409 
410 #ifdef Q_OS_WIN32
411  QWinTaskbarProgress *progress = taskbar_button_->progress();
412  progress->setVisible(need_progress);
413  progress->setMinimum(0);
414  progress->setMaximum(status.total);
415  progress->setValue(status.progress);
416  progress->setPaused(status.state == ProgramState::ItemsPaused);
417 #else
418  (void)need_progress;//Fix compilation warning(unused var on non-windows)
419 #endif
420 
421 }

423 bool MainWindow::eventFilter(QObject *o, QEvent *e) {
424  if (o == tab_bar_ && e->type() == QEvent::MouseButtonPress) {
425  QMouseEvent *mouse_event = static_cast<QMouseEvent *>(e);
426  if (mouse_event->button() == Qt::MidButton) {
427  // middle button pressed on a tab
428  int index = tab_bar_->tabAt(mouse_event->pos());
429  // remove tab and Search if it's not "+"
430  if (index >= 0 && index < tab_bar_->count() - 1) {
431  tab_bar_->removeTab(index);
432  auto search = searches_[index];
433  if (previous_search_ == search)
434  previous_search_ = nullptr;
435  if (current_search_ == search)
436  current_search_ = nullptr;
437  delete searches_[index];
438  searches_.erase(searches_.begin() + index);
439  if (static_cast<size_t>(tab_bar_->currentIndex()) == searches_.size())
440  tab_bar_->setCurrentIndex(searches_.size() - 1);
441  OnTabChange(tab_bar_->currentIndex());
442  // that's because after removeTab text will be set to previous search's caption
443  // which is because my way of dealing with "+" tab is hacky and should be replaced by something sane
444  tab_bar_->setTabText(tab_bar_->count() - 1, "+");
445  }
446  return true;
447  }
448  }
449  return QMainWindow::eventFilter(o, e);
450 }

477 void MainWindow::ModelViewRefresh() {
478  app_->buyout_manager().Save();
479 
480  // Save view properties if no search fields are populated
481  // AND we're viewing in Tab mode
482  if (previous_search_ && !previous_search_->IsAnyFilterActive()
483  && previous_search_->GetViewMode() == Search::ByTab)
484  previous_search_->SaveViewProperties();
485 
486  previous_search_ = current_search_;
487 
488  current_search_->Activate(app_->items_manager().items());
489 
490  ui->viewComboBox->setCurrentIndex(static_cast<int>(current_search_->GetViewMode()));
491 
492  connect(ui->treeView->selectionModel(), SIGNAL(currentChanged(const QModelIndex&, const QModelIndex&)),
493  this, SLOT(OnTreeChange(const QModelIndex&, const QModelIndex&)));
494 
495  ui->treeView->reset();
496  if (current_search_->IsAnyFilterActive() || current_search_->GetViewMode() == Search::ByItem) {
497  // Policy is to expand all tabs when any search fields are populated
498  // Also expand by default if we're in Item view mode
499  ExpandCollapse(TreeState::kExpand);
500  } else {
501  // Restore view properties if no search fields are populated AND current mode is tab mode
502  current_search_->RestoreViewProperties();
503  ResizeTreeColumns();
504  }
505  tab_bar_->setTabText(tab_bar_->currentIndex(), current_search_->GetCaption());
506 }

918 void MainWindow::OnUploadFinished() {
919  ui->uploadTooltipButton->setDisabled(false);
920  ui->uploadTooltipButton->setText("Upload to imgur");
921 
922  SelfDestructingReply reply(qobject_cast<QNetworkReply *>(QObject::sender()));
923  QByteArray bytes = reply->readAll();
924 
925  rapidjson::Document doc;
926  doc.Parse(bytes.constData());
927 
928  if (doc.HasParseError() || !doc.IsObject() || !doc.HasMember("status") || !doc["status"].IsNumber()) {
929  QLOG_ERROR() << "Imgur API returned invalid data (or timed out): " << bytes;
930  reply->deleteLater();
931  return;
932  }
933  if (doc["status"].GetInt() != 200) {
934  QLOG_ERROR() << "Imgur API returned status!=200: " << bytes;
935  reply->deleteLater();
936  return;
937  }
938  if (!doc.HasMember("data") || !doc["data"].HasMember("link") || !doc["data"]["link"].IsString()) {
939  QLOG_ERROR() << "Imgur API returned malformed reply: " << bytes;
940  reply->deleteLater();
941  return;
942  }
943  std::string url = doc["data"]["link"].GetString();
944  QApplication::clipboard()->setText(url.c_str());
945  QLOG_INFO() << "Image successfully uploaded, the URL is" << url.c_str() << "It also was copied to your clipboard.";

947  reply->deleteLater();
