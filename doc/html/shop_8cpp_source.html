84 void Shop::Update() {
85  if (submitting_) {
86  QLOG_WARN() << "Submitting shop right now, the request to update shop data will be ignored";
87  return;
88  }
89  shop_data_outdated_ = false;
90  shop_data_.clear();
91  std::string data = "";
92  std::vector<AugmentedItem> aug_items;
93  AugmentedItem tmp = AugmentedItem();
94  //Get all buyouts to be able to sort them
95  for (auto &item : app_.items_manager().items()) {
96  tmp.item = item.get();
97  tmp.bo = app_.buyout_manager().Get(*item);
98 
99  if (!tmp.bo.IsPostable())
100  continue;
101  if (item->location().socketed())
102  continue;
103  aug_items.push_back(tmp);
104  }
105  if (aug_items.size() == 0)
106  return;
107  std::sort(aug_items.begin(), aug_items.end());
108 
109  Buyout current_bo = aug_items[0].bo;
110  data += SpoilerBuyout(current_bo);
111  for (auto &aug : aug_items) {
112  if (aug.bo.type != current_bo.type || aug.bo.currency != current_bo.currency || aug.bo.value != current_bo.value) {
113  current_bo = aug.bo;
114  data += "[/spoiler]";
115  data += SpoilerBuyout(current_bo);
116  }
117  std::string item_string = aug.item->location().GetForumCode(app_.league());
118  if (data.size() + item_string.size() + shop_template_.size()+ kSpoilerOverhead + QString("[/spoiler]").size() > kMaxCharactersInPost) {
119  data +="[/spoiler]";
120  shop_data_.push_back(data);
121  data = SpoilerBuyout(current_bo);
122  data += item_string;
123  } else {
124  data += item_string;
125  }
126  }
127  if (!data.empty())
128  shop_data_.push_back(data);
129 
130  for (size_t i = 0; i < shop_data_.size(); ++i)
131  shop_data_[i] = Util::StringReplace(shop_template_, kShopTemplateItems, "[spoiler]" + shop_data_[i] + "[/spoiler]");
132 
133  shop_hash_ = Util::Md5(Util::StringJoin(shop_data_, ";"));
134 }

140 void Shop::SubmitShopToForum(bool force) {
141  if (submitting_) {
142  QLOG_WARN() << "Already submitting your shop.";
143  return;
144  }
145  if (threads_.empty()) {
146  QLOG_ERROR() << "Asked to update a shop with no shop ID defined.";
147  return;
148  }
149 
150  if (shop_data_outdated_)
151  Update();
152 
153  std::string previous_hash = app_.data().Get("shop_hash");
154  // Don't update the shop if it hasn't changed
155  if (previous_hash == shop_hash_ && !force)
156  return;
157 
158  if (threads_.size() < shop_data_.size()) {
159  QLOG_WARN() << "Need" << shop_data_.size() - threads_.size() << "more shops defined to fit all your items.";
160  }
161 
162  requests_completed_ = 0;
163  submitting_ = true;
164  SubmitSingleShop();
165 }

171 void Shop::SubmitSingleShop() {
172  CurrentStatusUpdate status = CurrentStatusUpdate();
173  status.state = ProgramState::ShopSubmitting;
174  status.progress = requests_completed_;
175  status.total = threads_.size();
176  if (requests_completed_ == threads_.size()) {
177  status.state = ProgramState::ShopCompleted;
178  submitting_ = false;
179  app_.data().Set("shop_hash", shop_hash_);
180  } else {
181  // first, get to the edit-thread page to grab CSRF token
182  QNetworkReply *fetched = app_.logged_in_nm().get(QNetworkRequest(QUrl(ShopEditUrl(requests_completed_).c_str())));
183  new QReplyTimeout(fetched, kEditThreadTimeout);
184  connect(fetched, SIGNAL(finished()), this, SLOT(OnEditPageFinished()));
185  }
186  emit StatusUpdate(status);
187 }

189 void Shop::OnEditPageFinished() {
190  QNetworkReply *reply = qobject_cast<QNetworkReply *>(QObject::sender());
191  QByteArray bytes = reply->readAll();
192  std::string page(bytes.constData(), bytes.size());
193  std::string hash = Util::GetCsrfToken(page, "hash");
194  if (hash.empty()) {
195  QLOG_ERROR() << "Can't update shop -- cannot extract CSRF token from the page. Check if thread ID is valid."
196  << "If you're using Steam to login make sure you use the same login method (steam or login/password) in Acquisition, Path of Exile website and Path of Exile game client."
197  << "For example, if you created a shop thread while using Steam to log into the website and then logged into Acquisition with login/password it will not work."
198  << "In this case you should either recreate your shop thread or use a correct login method in Acquisition.";
199  submitting_ = false;
200  return;
201  }
202 
203  // now submit our edit
204 
205  // holy shit give me some html parser library please
206  std::string title = Util::FindTextBetween(page, "<input type=\"text\" name=\"title\" id=\"title\" onkeypress=\"return&#x20;event.keyCode&#x21;&#x3D;13\" value=\"", "\">");
207  if (title.empty()) {
208  QLOG_ERROR() << "Can't update shop -- title is empty. Check if thread ID is valid.";
209  submitting_ = false;
210  return;
211  }
212 
213  QUrlQuery query;
214  query.addQueryItem("hash", hash.c_str());
215  query.addQueryItem("title", Util::Decode(title).c_str());
216  query.addQueryItem("content", requests_completed_ < shop_data_.size() ? shop_data_[requests_completed_].c_str() : "Empty");
217  query.addQueryItem("submit", "Submit");
218 
219  QByteArray data(query.query().toUtf8());
220  QNetworkRequest request((QUrl(ShopEditUrl(requests_completed_).c_str())));
221  request.setHeader(QNetworkRequest::ContentTypeHeader, "application/x-www-form-urlencoded");
222  QNetworkReply *submitted = app_.logged_in_nm().post(request, data);
223  new QReplyTimeout(submitted, kEditThreadTimeout);
224  connect(submitted, SIGNAL(finished()), this, SLOT(OnShopSubmitted()));
225 }

241 
242  ++requests_completed_;
