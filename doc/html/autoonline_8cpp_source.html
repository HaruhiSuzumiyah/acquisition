36 bool is_poe_running_locally() {
37 #if defined(Q_OS_LINUX) || defined(Q_OS_MAC)
38  QProcess process;
39  process.start("/bin/sh -c \"ps -ax|grep PathOfExile|grep -v grep|wc -l\"");
40  process.waitForFinished(-1);
41  QString i = process.readAllStandardOutput();
42  return i.toInt() > 0;
43 #elif defined(Q_OS_WIN)
44 // http://msdn.microsoft.com/en-us/library/windows/desktop/ms686701%28v=vs.85%29.aspx
45  HANDLE hProcessSnap;
46  PROCESSENTRY32 pe32;
47 
48  hProcessSnap = CreateToolhelp32Snapshot(TH32CS_SNAPPROCESS, 0);
49  if (hProcessSnap == INVALID_HANDLE_VALUE) {
50  QLOG_ERROR() << "CreateToolhelp32Snapshot (of processes)";
51  return false;
52  }
53 
54  pe32.dwSize = sizeof(PROCESSENTRY32);
55 
56  if (!Process32First(hProcessSnap, &pe32)) {
57  QLOG_ERROR() << "Process32First";
58  CloseHandle(hProcessSnap);
59  return false;
60  }
61 
62  bool found = false;
63  do {
64  QString s = QString::fromUtf16((char16_t*)pe32.szExeFile);
65  if (s.contains("PathOfExile"))
66  found = true;
67  } while(Process32Next(hProcessSnap, &pe32));
68 
69  CloseHandle(hProcessSnap);
70 
71  return found;
72 #endif
73 }

126 void AutoOnline::Check() {
127  bool running = is_poe_running_locally();
128  if (IsRemoteScriptSet()){
129  running = is_poe_running_remotely(process_script_);
130  } else {
131  running = is_poe_running_locally();
132  }
133 
134  if (running || previous_status_) {
135  SendOnlineUpdate(running);
136  }
137 
138  previous_status_ = running;
139 
140  emit Update(running);
141 }

145 void AutoOnline::SetUrl(const std::string &url) {
146  bool prefix_is_valid = false;
147  for (auto &valid_prefix : url_valid_prefixes) {
148  prefix_is_valid |= url.compare(0, valid_prefix.size(), valid_prefix) == 0;
149  }
150  if (!prefix_is_valid) {
151  QLOG_WARN() << "Online URL is probably invalid.";
152  }
153  url_ = url;
154  // remove trailing '/'s
155  while (url_.size() > 0 && url_[url_.size() - 1] == '/')
156  url_.erase(url_.size() - 1);
157  sensitive_data_.Set("online_url", url_);
158 }
